58/******************************************************************************
59 * Constructor
60 *
61 * @param pName filename
62 * @param pDataSize volume resolution
63 * @param pBlocksize brick resolution
64 * @param pBordersize brick border size
65 * @param pUseCache flag to tell wheter or not a cache mechanismn is required when reading files (nodes and bricks)
66 ******************************************************************************/
67template< typename TDataTypeList >
68GsDataLoader< TDataTypeList >
69::GsDataLoader( const std::string& pName, const uint3& pBlocksize, int pBordersize, bool pUseCache )
70: GsIDataLoader< TDataTypeList >()
71{
72 // First, read Meta Data file to retrieve all mandatory information
73 uint resolution = 0;
74 int parse = this->parseXMLFile( pName.c_str(), resolution );
75 assert( parse == 0 );
76 // TODO : handle error
77 // ...
78
79 // Fill member variables
80 this->_bricksRes.x = pBlocksize.x;
81 this->_bricksRes.y = pBlocksize.y;
82 this->_bricksRes.z = pBlocksize.z;
83 this->_numChannels = Loki::TL::Length< TDataTypeList >::value;
84 this->_volumeRes = make_uint3( resolution );//TO CHANGE
85 // LOG
86 printf( "%d\n", _volumeRes.x );
87 this->_borderSize = pBordersize;
88 this->_mipMapOrder = 2;
89 this->_useCache = pUseCache;
90
91 // Compute number of mipmaps levels
92 int dataResMin = mincc( this->_volumeRes.x, mincc( this->_volumeRes.y, this->_volumeRes.z ) );
93 int blocksNumLevels = static_cast< int >( log( static_cast< float >( this->_bricksRes.x ) ) / log( static_cast< float >( this->_mipMapOrder ) ) );
94 this->_numMipMapLevels = static_cast< int >( log( static_cast< float >( dataResMin ) ) / log( static_cast< float >( this->_mipMapOrder ) ) ) + 1 ;
95 this->_numMipMapLevels = this->_numMipMapLevels - blocksNumLevels;
96 if ( this->_numMipMapLevels < 1 )
97 {
98 this->_numMipMapLevels = 1;
99 }
100
101 // Compute full brick resolution (with borders)
102 uint3 true_bricksRes = this->_bricksRes + make_uint3( 2 * this->_borderSize );
103
104 // Build the list of all filenames that producer will have to load (nodes and bricks).
105 //this->makeFilesNames( pName.c_str() );
106
107 // If cache mechanismn is required, read all files (nodes and bricks),
108 // and store data in associated buffers.
109 if ( this->_useCache )
110 {
111 // Iterate through mipmap levels
112
113 for ( int level = 0; level < _numMipMapLevels; level++ )
114 {
115
116 // Retrieve node filename at current mipmap level
117 //
118 // Files are stored by mipmap level in the list :
119 // - first : node file
120 // - then : brick file for each channel
121 std::string fileNameIndex = _filesNames[ ( _numChannels + 1 ) * level ];
122
123 // Open node file
124 FILE* fileIndex = fopen( fileNameIndex.c_str(), "rb" );
125 if ( fileIndex )
126 {
127
128 // Retrieve node file size
129#ifdef WIN32
130 _fseeki64( fileIndex, 0, SEEK_END );
131
132 __int64 size = _ftelli64( fileIndex );
133 __int64 expectedSize = (__int64)powf( 8.0f, static_cast< float >( level ) ) * sizeof( unsigned int );
134#else
135 fseeko( fileIndex, 0, SEEK_END );
136
137 off_t size = ftello( fileIndex );
138 off_t expectedSize = (off_t)powf( 8.0f, static_cast< float >( level ) ) * sizeof( unsigned int );
139#endif
140 // Handle error
141 if ( size != expectedSize )
142 {
143 std::cerr << "GsDataLoader::GsDataLoader: file size expected = " << expectedSize
144 << ", size returned = " << size << " for " << fileNameIndex << std::endl;
145 }
146
147 // Allocate a buffer in which read node data will be stored
148 unsigned int* tmpCache = new unsigned int[ size / 4 ];
149
150 // Position file pointer at beginning of file
151#ifdef WIN32
152 _fseeki64( fileIndex, 0, SEEK_SET );
153#else
154 fseeko( fileIndex, 0, SEEK_SET );
155#endif
156 // Read node file and store data in the tmpCache buffer
157 if ( fread( tmpCache, 1, static_cast< size_t >( size ), fileIndex ) != size )
158 {
159 // Handle error if reading node file has failed
160 std::cout << "GsDataLoader::GsDataLoader: Unable to open file " << this->_filesNames[ level ] << std::endl;
161
162 this->_useCache = false;
163 }
164
165 // Close node file
166 fclose( fileIndex );
167
168 // Store node data in associated cache
169 _blockIndexCache.push_back( tmpCache );
170 }
171 else
172 {
173 // Handle error if opening node file has failed
174 std::cout << "GsDataLoader::GsDataLoader : Unable to open file index " << fileNameIndex << std::endl;
175 }
176
177 // For current mipmap level, iterate through channels
178 for ( size_t channel = 0; channel < _numChannels; channel++ )
179 {
180 // Retrieve brick filename at current channel (at current mipmap level)
181 //
182 // Files are stored by mipmap level in the list :
183 // - first : node file
184 // - then : brick file for each channel
185
186 // Open brick file
187 unsigned int fileIndex = ( _numChannels + 1 ) * level + channel + 1;
188 if ( fileIndex >= this->_filesNames.size() )
189 {
190 assert( false );
191 std::cout << "GsDataLoader::GsDataLoader() => File index error." << std::endl;
192 }
193 FILE* brickFile = fopen( this->_filesNames[ fileIndex ].c_str(), "rb" );
194 if ( brickFile )
195 {
196 // Retrieve brick file size
197#ifdef WIN32
198 _fseeki64( brickFile, 0, SEEK_END );
199
200 __int64 size = _ftelli64( brickFile );
201#else
202 fseeko( brickFile, 0, SEEK_END );
203
204 off_t size = ftello( brickFile );
205#endif
206 // Allocate a buffer in which read brick data will be stored
207 unsigned char* tmpCache;
208#if USE_GPUFETCHDATA
209 cudaHostAlloc( (void**)&tmpCache, size, cudaHostAllocMapped | cudaHostAllocWriteCombined );
210
211 void* deviceptr;
212 cudaHostGetDevicePointer( &deviceptr, tmpCache, 0 );
213 std::cout << "Device pointer host mem: " << static_cast< uint >( deviceptr ) << "\n";
214#else
215 // cudaMallocHost( (void**)&tmpCache, size ); // pinned memory
216 // cudaHostAlloc( (void **)&tmpCache, size );
217 tmpCache = new unsigned char[ static_cast< size_t >( size ) ];
218#endif
219 GV_CHECK_CUDA_ERROR( "GsDataLoader::GsDataLoader: cache alloc" );
220
221 // Position file pointer at beginning of file
222#ifdef WIN32
223 _fseeki64( brickFile, 0, SEEK_SET );
224#else
225 fseeko( brickFile, 0, SEEK_SET );
226#endif
227
228 // Read brick file and store data in the tmpCache buffer
229 if ( fread( tmpCache, 1, static_cast< size_t >( size ), brickFile ) != size )
230 {
231 // Handle error if reading brick file has failed
232 std::cout << "GsDataLoader::GsDataLoader: Can't read file" << std::endl;
233
234 this->_useCache = false;
235 }
236
237 // Close brick file
238 fclose( brickFile );
239
240 // Store brick data in associated cache
241 _blockCache.push_back( tmpCache );
242 }
243 else
244 {
245 // Handle error if opening brick file has failed
246 std::cout << "GsDataLoader::GsDataLoader: Unable to open file " << this->_filesNames[(_numChannels + 1) * level + channel + 1] << std::endl;
247
248 this->_useCache = false;
249 }
250 }
251 }
252 }

322/******************************************************************************
323 * Parses the XML configuration file
324 *
325 * @param pFilename the filename of the XML file
326 ******************************************************************************/
327template< typename TDataTypeList >
328int GsDataLoader< TDataTypeList >
329::parseXMLFile( const char* pFilename, uint& pResolution )
330{
331 // Reset internal state
332 pResolution = 1;
333 _filesNames.clear();
334
335 // Use GigaSpace Meta Data file reader
336 GvStructure::GsReader gigaSpaceReader;
337 // - read Meta Data file
338 bool statusOK = gigaSpaceReader.read( pFilename );
339 pResolution = gigaSpaceReader.getModelResolution();
340 _filesNames = gigaSpaceReader.getFilenames();
341 // Handle error(s)
342 assert( statusOK );
343 if ( ! statusOK )
344 {
345 std::cout << "Error during reading GigaSpace's Meta Data file" << std::endl;
346
347 return -1;
348 }
349
350 return 0;
351}

353/******************************************************************************
354 * Retrieve the node encoded address given a mipmap level and a 3D node indexed position
355 *
356 * @param pLevel mipmap level
357 * @param pBlockPos the 3D node indexed position
358 *
359 * @return the node encoded address
360 ******************************************************************************/
361template< typename TDataTypeList >
362unsigned int GsDataLoader< TDataTypeList >
363::getBlockIndex( int pLevel, const uint3& pBlockPos ) const
364{
365 // Retrieve the resolution at given level (i.e. the number of voxels in each dimension)
366 uint3 levelSize = getLevelRes( pLevel );
367
368 // Number of nodes at given level
369 uint3 blocksInLevel = levelSize / this->_bricksRes;
370
371 unsigned int indexValue = 0;
372
373 // Check whether or not, cache mechanism is used
374 if ( _useCache )
375 {
376 // _blockIndexCache is the buffer containing all read nodes data.
377 // This a 2D array containing, for each mipmap level, the list of nodes addresses.
378
379 // Compute the index of the node in the buffer of nodes, given its position
380 //
381 // Nodes are stored in increasing order from X axis first, then Y axis, then Z axis.
382 uint indexPos = pBlockPos.x + pBlockPos.y * blocksInLevel.x + pBlockPos.z * blocksInLevel.x * blocksInLevel.y;
383
384 // Get the node address
385 indexValue = _blockIndexCache[ pLevel ][ indexPos ];
386 }
387 else
388 {
389 // Compute the index of the node in the buffer of nodes, given its position
390 //
391 // Nodes are stored in increasing order from X axis first, then Y axis, then Z axis.
392#ifdef WIN32
393 __int64 indexPos = ( (__int64)pBlockPos.x + (__int64)( pBlockPos.y * blocksInLevel.x ) + (__int64)( pBlockPos.z * blocksInLevel.x * blocksInLevel.y ) ) * sizeof( unsigned int );
394#else
395 off_t indexPos = ( (off_t)pBlockPos.x + (off_t)( pBlockPos.y * blocksInLevel.x ) + (off_t)( pBlockPos.z * blocksInLevel.x * blocksInLevel.y ) ) * sizeof( unsigned int );
396#endif
397 // Retrieve node filename at given level
398 std::string fileNameIndex = this->_filesNames[ ( _numChannels + 1 ) * pLevel ];
399
400 // Open node file
401 FILE* fileIndex = fopen( fileNameIndex.c_str(), "rb" );
402 if ( fileIndex )
403 {
404 // Position file pointer at position corresponding to the requested node information
405#ifdef WIN32
406 _fseeki64( fileIndex, indexPos, 0 );
407#else
408 fseeko( fileIndex, indexPos, 0 );
409#endif
410 // Read node file and store requested node address in indexValue
411 if ( fread( &indexValue, sizeof( unsigned int ), 1, fileIndex ) != 1 )
412 {
413 // Handle error if reading node file has failed
414 std::cerr << "GsDataLoader<T>::getBlockIndex(): fread failed." << std::endl;
415 }
416
417 // Close node file
418 fclose( fileIndex );
419 }
420 else
421 {
422 // Handle error if opening node file has failed
423 std::cerr << "GsDataLoader<T>::getBlockIndex() : Unable to open file index "<<fileNameIndex << std::endl;
424 }
425 }
426
427 return indexValue;
428}

430/******************************************************************************
431 * Load a brick given a mipmap level, a 3D node indexed position,
432 * the data pool and an offset in the data pool.
433 *
434 * @param pLevel mipmap level
435 * @param pBlockPos the 3D node indexed position
436 * @param pDataPool the data pool
437 * @param pOffsetInPool offset in the data pool
438 *
439 * @return a flag telling wheter or not the brick has been loaded (some brick can contain no data).
440 ******************************************************************************/
441template< typename TDataTypeList >
442bool GsDataLoader< TDataTypeList >
443::loadBrick( int pLevel, const uint3& pBlockPos, GvCore::GPUPoolHost< GvCore::Array3D, TDataTypeList >* pDataPool, size_t pOffsetInPool )
444{
445 // Retrieve the resolution at given level (i.e. the number of voxels in each dimension)
446 uint3 levelSize = getLevelRes( pLevel );
447
448 //uint3 blocksInLevel = levelSize / this->_bricksRes; // seem to be not used anymore
449
450 // Compute full brick resolution (with borders)
451 uint3 trueBlocksRes = this->_bricksRes + make_uint3( 2 * this->_borderSize );
452
453 // Compute the brick size alignment in memory (with borders)
454 size_t blockMemSize = static_cast< size_t >( trueBlocksRes.x * trueBlocksRes.y * trueBlocksRes.z );
455
456 // Retrieve the node encoded address given a mipmap level and a 3D node indexed position
457 unsigned int indexVal = getBlockIndex( pLevel, pBlockPos );
458
459 // Test if node contains a brick
460 if ( indexVal & GV_VTBA_BRICK_FLAG )
461 {
462 // Use a channel initializer to read the brick
463 GsBrickLoaderChannelInitializer< TDataTypeList > channelInitializer( this, indexVal, blockMemSize, pLevel, pDataPool, pOffsetInPool );
464 GvCore::StaticLoop< GsBrickLoaderChannelInitializer< TDataTypeList >, Loki::TL::Length< TDataTypeList >::value - 1 >::go( channelInitializer );
465
466 return true;
467 }
468
469 return false;
470}

472/******************************************************************************
473 * Helper function used to determine the type of regions in the data structure.
474 * The data structure is made of regions containing data, empty or constant regions.
475 *
476 * Retrieve the node and associated brick located in this region of space,
477 * and depending of its type, if it contains data, load it.
478 *
479 * @param pPosition position of a region of space
480 * @param pSize size of a region of space
481 * @param pBrickPool data cache pool. This is where all data reside for each channel (color, normal, etc...)
482 * @param pOffsetInPool offset in the brick pool
483 *
484 * @return the type of the region (.i.e returns constantness information for that region)
485 ******************************************************************************/
486template< typename TDataTypeList >
487GsDataLoader< TDataTypeList >::VPRegionInfo GsDataLoader< TDataTypeList >
488::getRegion( const float3& pPosition, const float3& pSize, GvCore::GPUPoolHost< GvCore::Array3D, TDataTypeList >* pBrickPool, size_t pOffsetInPool )
489{
490 // Retrieve the level of resolution associated to a given size of a region of space.
491 int level = getDataLevel( pSize, _bricksRes );
492
493 // Retrieve the indexed coordinates associated to position of region of space at level of resolution.
494 uint3 coordsInLevel = getCoordsInLevel( level, pPosition );
495 // Retrieve the indexed coordinates of a block (i.e. a node) in the blocks grid of a region of space at a level of resolution.
496 uint3 blockCoords = getBlockCoords( level, pPosition );
497
498 // Check mipmap level bounds
499 if ( level >= 0 && level < _numMipMapLevels )
500 {
501 // Try to load brick given localization parameters
502 if ( loadBrick( level, blockCoords, pBrickPool, pOffsetInPool ) )
503 {
504 return GsDataLoader< TDataTypeList >::VP_UNKNOWN_REGION;
505 }
506 else
507 {
508 return GsDataLoader< TDataTypeList >::VP_CONST_REGION; // Correct ?
509 }
510 }
511 else
512 {
513 // Handle error
514 std::cout << "VolProducerBlocksOptim::getZone() : Invalid requested block dimentions" << std::endl;
515 return GsDataLoader< TDataTypeList >::VP_CONST_REGION;
516 }
517}

573/******************************************************************************
574 * Retrieve the node located in a region of space,
575 * and get its information (i.e. address containing its data type region).
576 *
577 * @param pPosition position of a region of space
578 * @param pSize size of a region of space
579 *
580 * @return the node encoded information

582template< typename TDataTypeList >
583uint GsDataLoader< TDataTypeList >
584::getRegionInfoNew( const float3& pPosition, const float3& pSize )
585{
586 // Retrieve the level of resolution associated to a given size of a region of space.
587 int level = getDataLevel( pSize, _bricksRes );
588 // Retrieve the indexed coordinates of a block (i.e. a node) in the blocks grid of a region of space at a level of resolution.
589 uint3 blockPosition = getBlockCoords( level, pPosition );
590
591 // Retrieve the resolution at given level (i.e. the number of voxels in each dimension)
592 uint3 levelsize = getLevelRes( level );
593 //uint3 blocksinlevel = levelsize / this->_bricksRes; // seem to be not used anymore...
594
595 // Compute full brick resolution (with borders)
596 uint3 trueBlocksRes = this->_bricksRes + make_uint3( 2 * this->_borderSize );
597
598 // Check mipmap level bounds
599 if ( level >= 0 && level < _numMipMapLevels )
600 {
601 // Get the node encoded address in the pool.
602 //
603 // Retrieve the node encoded address given a mipmap level and a 3D node indexed position
604 // Apply a mask on the two first bits to retrieve node information
605 // (the other 30 bits are for x,y,z address).

650/******************************************************************************
651 * Read a brick given parameters to retrieve data localization in brick files or in cache of bricks.
652 *
653 * @param pChannel channel index (i.e. color, normal, density, etc...)
654 * @param pIndexVal associated node encoded address of the node in which the brick resides
655 * @param pBlockMemSize brick size alignment in memory (with borders)
656 * @param pLevel mipmap level
657 * @param pData data array corresponding to given channel index in the data pool (i.e. bricks of voxels)
658 * @param pOffsetInPool offset in the data pool
659 ******************************************************************************/
660template< typename TDataTypeList >
661template< typename TChannelType >
662inline void GsDataLoader< TDataTypeList >
663::readBrick( int pChannel, unsigned int pIndexVal, unsigned int pBlockMemSize, unsigned int pLevel, GvCore::Array3D< TChannelType >* pData, size_t pOffsetInPool )
664{
665 // pIndexVal & 0x3FFFFFFFU : this expression corresponds to the last 30 bits of the 32 bits node encoded address
666 // These 30 bits corresponds to the address of the node on x,y,z axes.
667 //
668 // Compute the offset
669 unsigned int filePos = ( pIndexVal & 0x3FFFFFFFU ) * pBlockMemSize * sizeof( TChannelType );
670
671 // Check whether or not, cache mechanism is used
672 if ( _useCache )
673 {
674 if ( _blockCache[ pLevel * _numChannels + pChannel ] )
675 {
676 // Copy data from cache to the channel array of the data pool
677 memcpy( pData->getPointer( pOffsetInPool )/* destination */,
678 _blockCache[ pLevel * _numChannels + pChannel ] + filePos/* source */,
679 pBlockMemSize * sizeof( TChannelType ) /* number of bytes*/ );
680 }
681 }
682 else
683 {
684 // Open brick file
685 FILE* file = fopen( _filesNames[ pLevel * ( _numChannels + 1 ) + pChannel + 1 ].c_str(), "rb" );
686 if ( file )
687 {
688 // Position file pointer at position corresponding to the requested brick data
689#ifdef WIN32
690 _fseeki64( file, filePos, 0 );
691#else
692 fseeko( file, filePos, 0 );
693#endif
694 // Read brick file and store data in the channel array of the data pool
695 fread( pData->getPointer( pOffsetInPool ), sizeof( TChannelType ), pBlockMemSize, file );
696
697 // Close brick file
698 fclose( file );
699 }
700 }
701}
