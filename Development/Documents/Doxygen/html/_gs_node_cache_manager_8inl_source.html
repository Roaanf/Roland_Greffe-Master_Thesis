83/******************************************************************************
84 * Constructor
85 *
86 * @param pCachesize size of the cache
87 * @param pPageTableArray the array of elements that the cache has to managed
88 * @param pGraphicsInteroperability a flag used to map buffers to OpenGL graphics library
89 ******************************************************************************/
90template< typename TDataStructure >
91GsNodeCacheManager< TDataStructure >
92::GsNodeCacheManager( const uint3& pCachesize, PageTableArrayType* pPageTableArray, uint pGraphicsInteroperability )
93: GvCore::GsISerializable()
94, _cacheSize( pCachesize )
95, _policy( ePreventReplacingUsedElementsPolicy )
96{
97 // Compute elements cache size
98 _elemsCacheSize = _cacheSize / ElementRes::get();
99
100 // Page table initialization
101 _d_pageTableArray = pPageTableArray;
102 _pageTable = new PageTableType();
103
104 // Initialize the timestamp buffer
105 _d_TimeStampArray = new GvCore::GsLinearMemory< uint >( _elemsCacheSize, pGraphicsInteroperability );
106 _d_TimeStampArray->fill( 0 );
107
108 this->_numElements = _elemsCacheSize.x * _elemsCacheSize.y * _elemsCacheSize.z - _cNbLockedElements;
109
110 _numElemsNotUsed = _numElements;
111
112 // List of elements in the cache
113 _d_elemAddressList = new thrust::device_vector< uint >( this->_numElements );
114 _d_elemAddressListTmp = new thrust::device_vector< uint >( this->_numElements );
115
116 // Buffer of usage (masks of non-used and used elements in the current frame)
117 _d_TempMaskList = GsCacheManagerResources::getTempUsageMask1( static_cast< size_t >( this->_numElements ) );
118 _d_TempMaskList2 = GsCacheManagerResources::getTempUsageMask2( static_cast< size_t >( this->_numElements ) );
119
120 thrust::fill( _d_TempMaskList->begin(), _d_TempMaskList->end(), 0 );
121 thrust::fill( _d_TempMaskList2->begin(), _d_TempMaskList2->end(), 0 );
122
123 uint3 pageTableRes = _d_pageTableArray->getResolution();
124 uint pageTableResLinear = pageTableRes.x * pageTableRes.y * pageTableRes.z;
125
126 // Buffer of requests (masks of associated request in the current frame)
127 //_d_TempUpdateMaskList = GsCacheManagerResources::getTempUsageMask1( pageTableRes.x * pageTableRes.y *pageTableRes.z );
128 _d_TempUpdateMaskList = new thrust::device_vector< uint >( pageTableResLinear );
129 _d_UpdateCompactList = new thrust::device_vector< uint >( pageTableResLinear );
130
131 // Init
132 //
133 // Allocate size to speed up insert without need for push_back
134 thrust::host_vector< uint > tmpelemaddress( _elemsCacheSize.x * _elemsCacheSize.y * _elemsCacheSize.z );
135 uint3 pos;
136 uint index = 0;
137 for ( pos.z = 0; pos.z < _elemsCacheSize.z; pos.z++ )
138 for ( pos.y = 0; pos.y < _elemsCacheSize.y; pos.y++ )
139 for ( pos.x = 0; pos.x < _elemsCacheSize.x; pos.x++ )
140 {
141 tmpelemaddress[ index ] = GsNodeCacheManager< TDataStructure >::AddressType::packAddress( pos );
142 index++;
143 }
144 // Dont use element zero !
145 thrust::copy( /*first iterator*/tmpelemaddress.begin() + _cNbLockedElements, /*last iterator*/tmpelemaddress.end(), /*output iterator*/_d_elemAddressList->begin() );
146
147 uint cudppNumElem = std::max( pageTableRes.x * pageTableRes.y * pageTableRes.z, this->_numElements );
148 _scanplan = GsCacheManagerResources::getScanPlan( cudppNumElem );
149
150 GS_CUDA_SAFE_CALL( cudaMalloc( (void**) &_d_numElementsPtr, sizeof( size_t ) ) );
151
152 // The associated GPU side object receive a reference on the timestamp buffer
153 _d_cacheManagerKernel._timeStampArray = _d_TimeStampArray->getDeviceArray();
154
155 // LOG info
156 std::cout << "\nNode Cache Manager" << std::endl;
157 std::cout << "- cache size : " << _cacheSize << std::endl;
158 std::cout << "- elements cache size : " << _elemsCacheSize << std::endl;
159 std::cout << "- nb elements : " << this->_numElements << std::endl;
160 std::cout << "- page table's linear resolution : " << pageTableResLinear << std::endl;
161}

313::clearCache()
314{
315 // FIXME: do we really need to do it with a cuda kernel ?

319 thrust::host_vector< uint > tmpelemaddress( _elemsCacheSize.x * _elemsCacheSize.y * _elemsCacheSize.z );
320 uint3 pos;
321 uint index = 0;
322 for ( pos.z = 0; pos.z < _elemsCacheSize.z; pos.z++ )
323 for ( pos.y = 0; pos.y < _elemsCacheSize.y; pos.y++ )

334 thrust::fill( _d_TempMaskList->begin(), _d_TempMaskList->end(), 0 );
335 thrust::fill( _d_TempMaskList2->begin(), _d_TempMaskList2->end(), 0 );
336 CUDAPM_STOP_EVENT( gpucachemgr_clear_fillML )

338 // Clear the time-stamp buffer
339 CUDAPM_START_EVENT( gpucachemgr_clear_fillTimeStamp )
340 _d_TimeStampArray->fill( 0 );
341 CUDAPM_STOP_EVENT( gpucachemgr_clear_fillTimeStamp )
342}

344/******************************************************************************
345 * Update the list of available elements according to their timestamps.
346 * Unused and recycled elements will be placed first.
347 *
348 * @param manageUpdatesOnly ...
349 *
350 * @return the number of available elements
351 ******************************************************************************/
352template< typename TDataStructure >
353uint GsNodeCacheManager< TDataStructure >
354::updateTimeStamps( bool manageUpdatesOnly )
355{
356 if ( true || this->_lastNumLoads > 0 )
357 {
358 // uint numElemsNotUsed = 0;
359 _numElemsNotUsed = 0;
360
361 uint cacheNumElems = getNumElements();
362
363 // std:cout << "manageUpdatesOnly "<< (int)manageUpdatesOnly << "\n";
364
365 uint activeNumElems = cacheNumElems;
366 // TODO: re-enable manageUpdatesOnly !
367 /*if ( manageUpdatesOnly )
368 {
369 activeNumElems = this->_lastNumLoads;
370 }*/
371 uint inactiveNumElems = cacheNumElems - activeNumElems;
372
373 const uint nbElemToSort = activeNumElems;
374 if ( nbElemToSort > 0 )
375 {
376 uint sortingStartPos = 0;
377
378 // ---- [ 1 ] ---- 1st step
379 //
380 // Find "used" and "unused" elements during current frame in one single pass
381
382 CUDAPM_START_EVENT( cache_updateTimestamps_createMasks );
383
384 // Create masks in a single pass
385 dim3 blockSize( 64, 1, 1 );
386 //dim3 blockSize( 128, 1, 1 );
387 uint numBlocks = iDivUp( nbElemToSort, blockSize.x );
388 dim3 gridSize = dim3( std::min( numBlocks, 65535U ) , iDivUp( numBlocks, 65535U ), 1 );
389
390 // This kernel creates the usage mask list of used and non-used elements (in current rendering pass) in a single pass
391 //
392 // Note : Generate an error with CUDA 3.2
393 GvCache::GvKernel_NodeCacheManagerFlagTimeStamps<<< gridSize, blockSize >>>(
394 /*in*/nbElemToSort,
395 /*in*/thrust::raw_pointer_cast( &(*_d_elemAddressList)[ sortingStartPos ] ),
396 /*in*/_d_TimeStampArray->getPointer(),
397 /*out*/thrust::raw_pointer_cast( &(*_d_TempMaskList)[ 0 ] ), /*resulting mask list of non-used elements*/
398 /*out*/thrust::raw_pointer_cast( &(*_d_TempMaskList2)[ 0 ] ) ); /*resulting mask list of used elements*/
399 GV_CHECK_CUDA_ERROR( "GvKernel_NodeCacheManagerFlagTimeStamps" );
400
401 CUDAPM_STOP_EVENT( cache_updateTimestamps_createMasks );
402
403 // ---- [ 2 ] ---- 2nd step
404 //
405 // ...
406
407 thrust::device_vector< uint >::const_iterator elemAddressListFirst = _d_elemAddressList->begin();
408 thrust::device_vector< uint >::const_iterator elemAddressListLast = _d_elemAddressList->begin() + nbElemToSort;
409 thrust::device_vector< uint >::iterator elemAddressListTmpFirst = _d_elemAddressListTmp->begin();
410
411 // Stream compaction to collect non-used elements at the beginning
412 CUDAPM_START_EVENT( cache_updateTimestamps_threadReduc1 );
413
414 // Given an array d_in and an array of 1/0 flags in deviceValid, returns a compacted array in d_out of corresponding only the "valid" values from d_in.
415 //
416 // Takes as input an array of elements in GPU memory (d_in) and an equal-sized unsigned int array in GPU memory (deviceValid) that indicate which of those input elements are valid.
417 // The output is a packed array, in GPU memory, of only those elements marked as valid.
418 //
419 // Internally, uses cudppScan.
420 //
421 // Algorithmn : WRITE in temporary buffer _d_elemAddressListTmp, the compacted list of NON-USED elements from _d_elemAddressList
422 // - number of un-used is returned in _d_numElementsPtr
423 // - elements are written at the beginning of the array
424 cudppCompact( /*handle to CUDPPCompactPlan*/_scanplan,
425 /* OUT : compacted output */thrust::raw_pointer_cast( &(*_d_elemAddressListTmp)[ inactiveNumElems ] ),
426 /* OUT : number of elements valid flags in the d_isValid input array */_d_numElementsPtr,
427 /* input to compact */thrust::raw_pointer_cast( &(*_d_elemAddressList)[ sortingStartPos ] ),
428 /* which elements in input are valid */thrust::raw_pointer_cast( &(*_d_TempMaskList)[ 0 ] ), /*non-used elements*/
429 /* nb of elements in input */nbElemToSort );
430
431 // ---- [ 3 ] ---- 3rd step
432 //
433 // ...
434
435 // Get number of non-used elements
436 size_t numElemsNotUsedST;
437 GS_CUDA_SAFE_CALL( cudaMemcpy( &numElemsNotUsedST, _d_numElementsPtr, sizeof( size_t ), cudaMemcpyDeviceToHost ) );
438 _numElemsNotUsed = (uint)numElemsNotUsedST + inactiveNumElems;
439
440 //--------------------------------------------------------
441 // TO DO
442 //
443 // - optimization : if _numElemsNotUsed is equal to max elements in cache, exit o avoid launching all following kernels
444 //--------------------------------------------------------
445
446 CUDAPM_STOP_EVENT( cache_updateTimestamps_threadReduc1 );
447
448 // ---- [ 4 ] ---- 4th step
449 //
450 // ...
451
452 // Stream compaction to collect used elements at the end
453 CUDAPM_START_EVENT( cache_updateTimestamps_threadReduc2 );
454
455 // Given an array d_in and an array of 1/0 flags in deviceValid, returns a compacted array in d_out of corresponding only the "valid" values from d_in.
456 //
457 // Takes as input an array of elements in GPU memory (d_in) and an equal-sized unsigned int array in GPU memory (deviceValid) that indicate which of those input elements are valid.
458 // The output is a packed array, in GPU memory, of only those elements marked as valid.
459 //
460 // Internally, uses cudppScan.
461 //
462 // Algorithmn : WRITE in temporary buffer _d_elemAddressListTmp, the compacted list of USED elements from _d_elemAddressList
463 // - elements are written at the end of the array, i.e. after the previous NON-USED ones
464 cudppCompact( /*handle to CUDPPCompactPlan*/_scanplan,
465 /* OUT : compacted output */thrust::raw_pointer_cast( &(*_d_elemAddressListTmp)[ _numElemsNotUsed ] ),
466 /* OUT : number of elements valid flags in the d_isValid input array */_d_numElementsPtr,
467 /* input to compact */thrust::raw_pointer_cast( &(*_d_elemAddressList)[ sortingStartPos ] ),
468 /* which elements in input are valid */thrust::raw_pointer_cast( &(*_d_TempMaskList2)[ 0 ] ),
469 /* nb of elements in input */nbElemToSort );
470
471 CUDAPM_STOP_EVENT( cache_updateTimestamps_threadReduc2 );
472 }
473 else
474 {
475 _numElemsNotUsed = cacheNumElems;
476 }
477
478 // ---- [ 5 ] ---- 5th step
479 //
480 // ...
481
482 // Swap buffers
483 //
484 // Temporary buffer "_d_elemAddressListTmp" where non-used elements are the beginning and used elements at the end,
485 // is swapped with _d_elemAddressList
486 thrust::device_vector< uint >* tmpl = _d_elemAddressList;
487 _d_elemAddressList = _d_elemAddressListTmp;
488 _d_elemAddressListTmp = tmpl;
489
490 this->_lastNumLoads = 0;
491 }
492
493 return _numElemsNotUsed;
494}

496/******************************************************************************
497 * Main method to launch the production of nodes or bricks
498 *
499 * @param updateList global buffer of requests of used elements only (node subdivision an brick lod/produce)
500 * @param numUpdateElems ...
501 * @param updateMask Type of request to handle (node subdivision or brick load/produce)
502 * @param maxNumElems Max number of elements to process
503 * @param numValidNodes ...
504 * @param gpuPool pool used to write new produced data inside (node pool or data pool)
505 *
506 * @return the number of produced elements
507 ******************************************************************************/
508template< typename TDataStructure >
509template< typename GPUPoolType, typename TProducerType >
510uint GsNodeCacheManager< TDataStructure >
511::genericWrite( uint* updateList, uint numUpdateElems, uint updateMask,
512 uint maxNumElems, uint numValidNodes, const GPUPoolType& gpuPool, TProducerType* pProducer )
513{
514 assert( pProducer != NULL );
515 // TO DO : check pProducer at run-time
516
517 uint numElems = 0;
518
519 if ( numUpdateElems > 0 )
520 {
521 const uint cacheId = 0;
522
523 // ---- [ 1 ] ---- 1st step
524 //
525 // Fill the buffer of mask of requests
526
527 CUDAPM_START_EVENT_CHANNEL( 0, cacheId, gpucache_nodes_manageUpdates );
528
529 // The "_d_UpdateCompactList" buffer is filled with only elements addresses that have a type of request equal to "updateMask"
530 // - it returns the number of elements concerned
531 numElems = createUpdateList( updateList, numUpdateElems, updateMask );
532
533 CUDAPM_STOP_EVENT_CHANNEL( 0, cacheId, gpucache_nodes_manageUpdates );
534
535 // Prevent loading more than the cache size
536 numElems = std::min( numElems, getNumElements() );
537
538 // Warning to prevent user that there are no more avalaible slots
539 if ( numElems > _numElemsNotUsed )
540 {
541 // LOG info
542 std::cout << "CacheManager< " << cacheId << " >: Warning: " << _numElemsNotUsed << " non-used slots available, but ask for : " << numElems << std::endl;
543 }
544
545 // Handle cache policy
546 if ( this->_policy & ePreventReplacingUsedElementsPolicy )
547 {
548 // Prevent replacing elements in use
549 numElems = std::min( numElems, _numElemsNotUsed );
550 }
551 if ( this->_policy & eSmoothLoadingPolicy )
552 {
553 // Smooth loading
554 numElems = std::min( numElems, maxNumElems );
555 }
556
557 // Check if we have requests to handle
558 if ( numElems > 0 )
559 {
560 // std::cout << "CacheManager<" << cacheId << ">: " << numElems << " requests" << std::endl;
561
562 // ---- [ 2 ] ---- 2nd step
563 //
564 // Invalidation phase
565
566 // Update internal counter
567 _totalNumLoads += numElems;
568 _lastNumLoads = numElems;
569
570 /* std::cout << "\t_totalNumLoads : " << _totalNumLoads << std::endl;
571 std::cout << "\t_lastNumLoads : " << _lastNumLoads << std::endl;
572 std::cout << "\t_numElemsNotUsed : " << _numElemsNotUsed << std::endl;*/
573
574 invalidateElements( numElems, numValidNodes ); // WARNING !!!! numElems a été modifié auparavant !!!! ===> ERREUR !!!!!!!!!!!!!!
575
576 // ---- [ 3 ] ---- 3rd step
577 //
578 // Write new elements into the cache
579
580 CUDAPM_START_EVENT_CHANNEL( 0, cacheId, gpucache_nodes_subdivKernel );
581
582 // Buffer of nodes addresses with their requests
583 thrust::device_vector< uint >* nodesAddressCompactList = _d_UpdateCompactList;
584 // Sorted buffer of elements (according to timestamps : "un-used ones" first then "used ones" after)
585 thrust::device_vector< uint >* elemsAddressCompactList = _d_elemAddressList;
586
587 // Ask provider to produce data
588 //
589 // This method is called by the cache manager when you have to produce data for a given pool
590 //
591 // numElems : number of elements to produce
592 // nodesAddressCompactList : list containing the addresses of the "numElems" nodes concerned
593 // elemsAddressCompactList : list containing "numElems" addresses where store the result
594 // gpuPool : pool for which we need to produce elements (node pol or data pool)
595 // _pageTable : page table associated to the pool (used to retrieve node's localization info [code and depth])
596 //_provider->template produceData< ElementRes >( numElems, nodesAddressCompactList, elemsAddressCompactList, gpuPool, _pageTable );
597 pProducer->produceData( numElems, nodesAddressCompactList, elemsAddressCompactList, Loki::Int2Type< 0 >() );
598
599 CUDAPM_STOP_EVENT_CHANNEL( 0, cacheId, gpucache_nodes_subdivKernel );
600 }
601 }
602
603 return numElems;
604}

606/******************************************************************************
607 * Create the "update" list of a given type.
608 *
609 * "Update list" is the list of elements and their associated requests of a given type (node subdivision or brick load/produce)
610 *
611 * @param inputList Buffer of node addresses and their associated requests. First two bits of their 32 bit addresses stores the type of request
612 * @param inputNumElem Number of elements to process
613 * @param pTestFlag type of request (node subdivision or brick load/produce)
614 *
615 * @return the number of requests of given type
616 ******************************************************************************/
617template< typename TDataStructure >
618uint GsNodeCacheManager< TDataStructure >
619::createUpdateList( uint* inputList, uint inputNumElem, uint pTestFlag )
620{
621 // ---- [ 1 ] ---- 1st step
622 //
623 // Fill the buffer of masks of valid elements whose attribute is equal to "pTestFlag"
624 // Result is placed in "_d_TempUpdateMaskList"
625
626 CUDAPM_START_EVENT( gpucachemgr_createUpdateList_createMask );
627
628 // Set kernel execution configuration
629 dim3 blockSize( 64, 1, 1 );
630 uint numBlocks = iDivUp( inputNumElem, blockSize.x );
631 dim3 gridSize = dim3( std::min( numBlocks, 65535U ) , iDivUp( numBlocks,65535U ), 1 );
632
633 // Given a flag indicating a type of request (subdivision or load), and the updated global buffer of requests,
634 // it fills a resulting mask buffer.
635 // In fact, this mask indicates, for each element of the usage list, if it was used in the current rendering pass
636 CacheManagerCreateUpdateMask<<< gridSize, blockSize, 0 >>>( inputNumElem, inputList, /*output*/(uint*)thrust::raw_pointer_cast( &(*_d_TempUpdateMaskList)[ 0 ] ), pTestFlag );
637 GV_CHECK_CUDA_ERROR( "UpdateCreateSubdivMask" );
638
639 CUDAPM_STOP_EVENT( gpucachemgr_createUpdateList_createMask );
640
641 // ---- [ 2 ] ---- 2nd step
642 //
643 // Concatenate only previous valid elements from input data in "inputList" into the buffer of requests
644 // Result is placed in "_d_UpdateCompactList"
645
646 CUDAPM_START_EVENT( gpucachemgr_createUpdateList_elemsReduction );
647
648 // Stream compaction
649 cudppCompact( _scanplan,
650 /*output*/thrust::raw_pointer_cast( &(*_d_UpdateCompactList)[ 0 ] ), /*nbValidElements*/_d_numElementsPtr,
651 /*input*/inputList, /*isValid*/(uint*)thrust::raw_pointer_cast( &(*_d_TempUpdateMaskList)[ 0 ] ),
652 /*nbElements*/inputNumElem );
653 GV_CHECK_CUDA_ERROR( "cudppCompact" );
654
655 // Get number of elements
656 size_t numElems;
657 GS_CUDA_SAFE_CALL( cudaMemcpy( &numElems, _d_numElementsPtr, sizeof( size_t ), cudaMemcpyDeviceToHost ) );
658
659 CUDAPM_STOP_EVENT( gpucachemgr_createUpdateList_elemsReduction );
660
661 return static_cast< uint >( numElems );
662}

664/******************************************************************************
665 * Invalidate elements
666 *
667 * Timestamps are reset to 1 and node addresses to 0 (but not the 2 first flags)
668 *
669 * @param numElems number of elements to invalidate (this is done for the numElems first elements, I.e. the unused ones)
670 * @param numValidPageTableSlots ...
671 ******************************************************************************/
672template< typename TDataStructure >
673void GsNodeCacheManager< TDataStructure >
674::invalidateElements( uint numElems, int numValidPageTableSlots )
675{
676 // ---- [ 1 ] ---- 1st step
677
678 // Invalidation procedure
679 {
680 // Set kernel execution configuration
681 dim3 blockSize( 64, 1, 1 );
682 uint numBlocks = iDivUp( numElems, blockSize.x );
683 dim3 gridSize = dim3( std::min( numBlocks, 65535U ), iDivUp( numBlocks, 65535U ), 1 );
684
685 // Reset the time stamp info of given elements to 1
686 uint* d_sortedElemAddressList = thrust::raw_pointer_cast( &(*_d_elemAddressList)[ 0 ] );
687 CacheManagerFlagInvalidations< ElementRes, GsNodeCacheManager< TDataStructure >::AddressType >
688 <<< gridSize, blockSize, 0 >>>( _d_cacheManagerKernel/*output : time stamps*/, numElems, /*input*/d_sortedElemAddressList );
689
690 GV_CHECK_CUDA_ERROR( "CacheManagerFlagInvalidations" );
691 }
692
693 // ---- [ 2 ] ---- 2nd step
694
695 {
696 uint3 pageTableRes = _d_pageTableArray->getResolution();
697 uint numPageTableElements = pageTableRes.x * pageTableRes.y * pageTableRes.z;
698 if ( numValidPageTableSlots >= 0 )
699 {
700 numPageTableElements = min( numPageTableElements, numValidPageTableSlots );
701 }
702
703 // Set kernel execution configuration
704 dim3 blockSize( 64, 1, 1 );
705 uint numBlocks = iDivUp( numPageTableElements, blockSize.x );
706 dim3 gridSize = dim3( std::min( numBlocks, 65535U ), iDivUp( numBlocks, 65535U ), 1 );
707
708 // Reset all node addresses in the cache to NULL (i.e 0).
709 // Only the first 30 bits of address are set to 0, not the 2 first flags.
710 CacheManagerInvalidatePointers< ElementRes, GsNodeCacheManager< TDataStructure >::AddressType >
711 <<< gridSize, blockSize, 0 >>>( /*in*/_d_cacheManagerKernel, /*in*/numPageTableElements, /*modified*/_d_pageTableArray->getDeviceArray() );
712
713 GV_CHECK_CUDA_ERROR( "VolTreeInvalidateNodePointers" );
714 }
715}
