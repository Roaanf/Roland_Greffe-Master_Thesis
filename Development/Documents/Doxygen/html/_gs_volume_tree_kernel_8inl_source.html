53/******************************************************************************
54 * Sample data in specified channel at a given position.
55 * 3D texture are used with hardware tri-linear interpolation.
56 *
57 * @param pBrickPos Brick position in the pool of bricks
58 * @param pPosInBrick Position in brick
59 *
60 * @return the sampled value
61 ******************************************************************************/
62template< class DataTList, class NodeTileRes, class BrickRes, uint BorderSize >
63template< int TChannel >
64__device__
65__forceinline__ float4 VolumeTreeKernel< DataTList, NodeTileRes, BrickRes, BorderSize >
66::getSampleValueTriLinear( float3 pBrickPos, float3 pPosInBrick ) const
67{
68 // Type definition of the data channel type
69 typedef typename GvCore::DataChannelType< DataTList, TChannel >::Result ChannelType;
70
71 float4 result = make_float4( 0.0f );
72 float3 samplePos = pBrickPos + pPosInBrick;
73
74 // Sample data in texture according to its type (float or not)
75 if ( ! ( GvCore::IsFloatFormat< ChannelType >::value ) )
76 {
77 gpuPoolTexFetch( TEXDATAPOOL, TChannel, 3, ChannelType, cudaReadModeNormalizedFloat, samplePos, result );
78 }
79 else
80 {
81 gpuPoolTexFetch( TEXDATAPOOL, TChannel, 3, ChannelType, cudaReadModeElementType, samplePos, result );
82 }
83
84 return result;
85}

87/******************************************************************************
88 * Sample data in specified channel at a given position.
89 * 3D texture are used with hardware tri-linear interpolation.
90 *
91 * @param mipMapInterpCoef mipmap interpolation coefficient
92 * @param brickChildPosInPool brick child position in pool
93 * @param brickParentPosInPool brick parent position in pool
94 * @param posInBrick position in brick
95 * @param coneAperture cone aperture
96 *
97 * @return the sampled value
98 ******************************************************************************/
99// QUESTION : le paramètre "coneAperture" ne semble pas utilisé ? A quoi sert-il (servait ou servira) ?
100template< class DataTList, class NodeTileRes, class BrickRes, uint BorderSize >
101template< int TChannel >
102__device__
103__forceinline__ float4 VolumeTreeKernel< DataTList, NodeTileRes, BrickRes, BorderSize >
104::getSampleValueQuadriLinear( float mipMapInterpCoef, float3 brickChildPosInPool, float3 brickParentPosInPool, float3 posInBrick, float coneAperture ) const
105{
106 float3 samplePos0 = posInBrick;
107 float3 samplePos1 = posInBrick / NodeTileRes::getFloat3();//* 0.5f;
108
109 float4 vox0, vox1;
110
111 // Sample data in brick
112 vox1 = getSampleValueTriLinear< TChannel >( brickParentPosInPool, samplePos1 );
113
114 if ( mipMapInterpCoef <= 1.0f )
115 {
116 // Sample data in child brick
117 vox0 = getSampleValueTriLinear< TChannel >( brickChildPosInPool, samplePos0 );
118
119 // Linear interpolation of results
120 vox1 = lerp( vox0, vox1, mipMapInterpCoef );
121 }
122
123 return vox1;
124}

126/******************************************************************************
127 * Sample data in specified channel at a given position.
128 * 3D texture are used with hardware tri-linear interpolation.
129 *
130 * @param mipMapInterpCoef mipmap interpolation coefficient
131 * @param brickChildPosInPool brick child position in pool
132 * @param brickParentPosInPool brick parent position in pool
133 * @param posInBrick position in brick
134 * @param coneAperture cone aperture
135 *
136 * @return the sampled value
137 ******************************************************************************/
138template< class DataTList, class NodeTileRes, class BrickRes, uint BorderSize >
139template< int TChannel >
140__device__
141__forceinline__ float4 VolumeTreeKernel< DataTList, NodeTileRes, BrickRes, BorderSize >
142::getSampleValue( float3 brickChildPosInPool, float3 brickParentPosInPool, float3 sampleOffsetInBrick, float coneAperture, bool mipMapOn, float mipMapInterpCoef ) const
143{
144 float4 vox;
145
146 // Sample data in texture
147 if ( mipMapOn && mipMapInterpCoef > 0.0f )
148 {
149 vox = getSampleValueQuadriLinear< TChannel >( mipMapInterpCoef,
150 brickChildPosInPool, brickParentPosInPool, sampleOffsetInBrick,
151 coneAperture );
152 }
153 else
154 {
155 vox = getSampleValueTriLinear< TChannel >( brickChildPosInPool, sampleOffsetInBrick );
156 }
157
158 return vox;
159}
