125 _nodesCacheManager->_pageTable->getKernel().childArray = _dataStructure->_childArray->getDeviceArray();
126 _nodesCacheManager->_pageTable->getKernel().locCodeArray = _dataStructure->_localizationCodeArray->getDeviceArray();
127 _nodesCacheManager->_pageTable->getKernel().locDepthArray = _dataStructure->_localizationDepthArray->getDeviceArray();
128//#ifndef GS_USE_MULTI_OBJECTS

137
138 // Data cache manager initialization
139 _bricksCacheManager->_pageTable->locCodeArray = _dataStructure->_localizationCodeArray;
140 _bricksCacheManager->_pageTable->locDepthArray = _dataStructure->_localizationDepthArray;
141 _bricksCacheManager->_pageTable->getKernel().childArray = _dataStructure->_childArray->getDeviceArray();
142 _bricksCacheManager->_pageTable->getKernel().dataArray = _dataStructure->_dataArray->getDeviceArray();
143 _bricksCacheManager->_pageTable->getKernel().locCodeArray = _dataStructure->_localizationCodeArray->getDeviceArray();
144 _bricksCacheManager->_pageTable->getKernel().locDepthArray = _dataStructure->_localizationDepthArray->getDeviceArray();
145 _bricksCacheManager->_totalNumLoads = 0;

157 cacheManagerMaxNbElements = std::max( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z, cacheManagerMaxNbElements );

159 GsCompute::GsDataParallelPrimitives::get().initializeSortPlan( cacheManagerMaxNbElements );

161 // Request buffers initialization
162 _updateBufferArray = new GvCore::GsLinearMemory< uint >( _nodePoolRes, graphicsInteroperability );
163 _updateBufferCompactList = new GvCore::GsLinearMemory< uint >( _nodePoolRes );
164
165 // Priority buffers (for production requests)
166 _priorityBufferArray = new GvCore::GsLinearMemory< int >( _nodePoolRes, graphicsInteroperability );
167 _priorityBufferCompactList = new GvCore::GsLinearMemory< uint >( _nodePoolRes );

169#ifdef GS_USE_MULTI_OBJECTS

171 _objectIDBufferCompactList = new GvCore::GsLinearMemory< uint >( _nodePoolRes );
172// _nodesCacheManager->_d_ObjectIDs = _objectIDBufferCompactList;
173// _bricksCacheManager->_d_ObjectIDs = _objectIDBufferCompactList;
174#endif
175
176 _totalNumBricksLoaded = 0;
177
178 // Device-side cache manager initialization
179 _dataProductionManagerKernel._updateBufferArray = this->_updateBufferArray->getDeviceArray();
180 _dataProductionManagerKernel._nodeCacheManager = this->_nodesCacheManager->getKernelObject();
181 _dataProductionManagerKernel._brickCacheManager = this->_bricksCacheManager->getKernelObject();
182 _dataProductionManagerKernel._priorityBufferArray = this->_priorityBufferArray->getDeviceArray();
183#ifdef GS_USE_MULTI_OBJECTS
184 _dataProductionManagerKernel._objectIDBuffer = this->_objectIDBuffer->getDeviceArray();
185#endif
186
187 // Initialize max number of requests the cache has to handle
188 _maxNbNodeSubdivisions = 5000;
189 _maxNbBrickLoads = 3000;
190 this->_nbNodeSubdivisionRequests = 0;
191 this->_nbBrickLoadRequests = 0;
192
193#ifndef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_DATAPRODUCTIONMANAGER
194 GS_CUDA_SAFE_CALL( cudaMalloc( (void**) &_d_nbValidRequests, sizeof( size_t ) ) );
195#else
196 #ifndef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_WITH_COMBINED_COPY
197 GS_CUDA_SAFE_CALL( cudaMalloc( (void**) &_d_nbValidRequests, sizeof( size_t ) ) );
198 #else
199 GS_CUDA_SAFE_CALL( cudaMalloc( (void**) &_d_nbValidRequests, 3 * sizeof( size_t ) ) );
200 _nodesCacheManager->_d_nbValidRequests = _d_nbValidRequests;
201 _bricksCacheManager->_d_nbValidRequests = _d_nbValidRequests;
202 #endif
203#endif
204
205 _d_validRequestMasks = new GvCore::GsLinearMemory< uint >( _nodePoolRes, graphicsInteroperability );
206
207 // TO DO : do lazy evaluation => ONLY ALLOCATE WHEN REQUESTED AND USED + free memory just after ?
208 _leafNodes = new thrust::device_vector< uint >( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z );
209 _emptyNodeVolume = new thrust::device_vector< float >( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z );
210
211 cudaEventCreate( &_startProductionNodes );
212 cudaEventCreate( &_stopProductionNodes );
213 cudaEventCreate( &_startProductionBricks );
214 cudaEventCreate( &_stopProductionBricks );
215}

290/******************************************************************************
291 * This method is called after the rendering process. She's responsible for processing requests.
292 *
293 * @return the number of requests processed.
294 *
295 * @todo Check whether or not the inversion call of updateTimeStamps() with manageUpdates() has side effects
296 ******************************************************************************/
297template< typename TDataStructure, typename TPriorityPoliciesManager >
298uint GsDataProductionManager< TDataStructure, TPriorityPoliciesManager >
299::handleRequests()
300{
301 // Measure the time taken by the last production.
302 if ( _lastProductionTimed
303 && ( _nbBrickLoadRequests != 0 || _nbNodeSubdivisionRequests != 0 ) )
304 {
305 cudaEventSynchronize( _stopProductionBricks );
306 float lastProductionNodesTime, lastProductionBricksTime;
307
308 cudaEventElapsedTime( &lastProductionNodesTime, _startProductionNodes, _stopProductionNodes );
309 cudaEventElapsedTime( &lastProductionBricksTime, _startProductionBricks, _stopProductionBricks );
310
311 // Don't take too low number of requests into account (in this cases, the additional
312 // costs of launching the kernel, compacting the array... is greater than the
313 // brick/node production time)
314 if ( _nbNodeSubdivisionRequests > 63 ) // hard-coded value...
315 {
316 _totalProducedNodes += _nbNodeSubdivisionRequests;
317 _totalNodesProductionTime += lastProductionNodesTime;
318 }
319
320 if ( _nbBrickLoadRequests > 63 ) // hard-coded value...
321 {
322 _totalProducedBricks += _nbBrickLoadRequests;
323 _totalBrickProductionTime += lastProductionBricksTime;
324 }
325
326 // Update the vector of statistics.
327 struct GsProductionStatistics stats;
328 //stats._frameId = ??? TODO
329 stats._nNodes = _nbNodeSubdivisionRequests;
330 stats._nodesProductionTime = lastProductionNodesTime;
331 stats._nBricks = _nbBrickLoadRequests;
332 stats._bricksProductionTime = lastProductionBricksTime;
333 _productionStatistics.push_back( stats ); // BEWARE : stack limit
334 }
335 // _isProductionTimeLimited should not be used inside this function since it can be changed
336 // by the user at anytime and we need to have a constant value throughout the whole function.
337 _lastProductionTimed = _isProductionTimeLimited;
338
339 // TO DO
340 // Check whether or not the inversion call of updateTimeStamps() with manageUpdates() has side effects
341
342 // Generate the requests buffer
343 //
344 // Collect and compact update informations for both nodes and bricks
345 /*CUDAPM_START_EVENT( dataProduction_manageRequests );
346 uint nbRequests = manageUpdates();
347 CUDAPM_STOP_EVENT( dataProduction_manageRequests );*/
348
349 // Stop post-render pass if no request
350 //if ( nbRequests > 0 )
351 //{
352 // Update time stamps
353 //
354 // - TO DO : the update time stamps could be done in parallel for nodes and bricks using streams
355 CUDAPM_START_EVENT( cache_updateTimestamps );
356 updateTimeStamps();
357 CUDAPM_STOP_EVENT( cache_updateTimestamps );
358
359 // Manage requests
360 //
361 // - TO DO : if updateTimestamps is before, this task could also be done in parallel using streams
362 CUDAPM_START_EVENT( dataProduction_manageRequests );
363 uint nbRequests = manageUpdates();
364 CUDAPM_STOP_EVENT( dataProduction_manageRequests );
365
366#ifdef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_DATAPRODUCTIONMANAGER
367 // Get number of elements
368 // BEWARE : synchronization to avoid an expensive final call to cudaDeviceSynchronize()
369 #ifndef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_WITH_COMBINED_COPY
370 size_t nbElementsTemp;
371 _nodesCacheManager->updateTimeStampsCopy( _intraFramePass );
372 _bricksCacheManager->updateTimeStampsCopy( _intraFramePass );
373 GS_CUDA_SAFE_CALL( cudaMemcpy( &nbElementsTemp, _d_nbValidRequests, sizeof( size_t ), cudaMemcpyDeviceToHost ) );
374 nbRequests = static_cast< uint >( nbElementsTemp );
375 #else
376 size_t nbElementsTemp[ 3 ];
377 GS_CUDA_SAFE_CALL( cudaMemcpy( nbElementsTemp, _d_nbValidRequests, 3 * sizeof( size_t ), cudaMemcpyDeviceToHost ) );
378 nbRequests = static_cast< uint >( nbElementsTemp[ 0 ] );
379 // BEWARE : in nodes/bricks managers, real value should be _numElemsNotUsed = (uint)numElemsNotUsedST + inactiveNumElems
380 _nodesCacheManager->_numElemsNotUsedST = static_cast< uint >( nbElementsTemp[ 1 ] );
381 _bricksCacheManager->_numElemsNotUsedST = static_cast< uint >( nbElementsTemp[ 2 ] );
382 #endif
383 // Launch final "stream compaction" steps for "used" elements
384 this->_numNodeTilesNotInUse = _nodesCacheManager->updateTimeStampsFinal( _intraFramePass );
385 this->_numBricksNotInUse = _bricksCacheManager->updateTimeStampsFinal( _intraFramePass );
386#endif
387
388 // Handle requests :
389#ifndef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_PRODUCER
390
391 // [ 1 ] - Handle the "subdivide nodes" requests
392
393 // Limit production according to the time limit.
394 // First, do as if all the requests were node subdivisions
395 if ( _lastProductionTimed )
396 {
397 if ( _totalProducedNodes != 0u )
398 {
399 nbRequests = min(
400 nbRequests,
401 static_cast< uint >( _productionTimeLimit * _totalProducedNodes / _totalNodesProductionTime ) );
402 }
403 cudaEventRecord( _startProductionNodes );
404 }
405
406 CUDAPM_START_EVENT( producer_nodes );
407 _nbNodeSubdivisionRequests = manageNodeProduction( nbRequests );
408 CUDAPM_STOP_EVENT( producer_nodes );
409
410 if ( _lastProductionTimed )
411 {
412 cudaEventRecord( _stopProductionNodes );
413 cudaEventRecord( _startProductionBricks );
414 }
415
416 // [ 2 ] - Handle the "load/produce bricks" requests
417
418 // Now, we know how many requests are node and how many are bricks, we can limit
419 // the number of bricks requests according to the number of node requests performed.
420 uint nbBricks = nbRequests;
421 if ( _lastProductionTimed && _totalProducedNodes != 0 && _totalProducedBricks != 0 )
422 {
423 // Evaluate how much time will be left after nodes subdivision
424 float remainingTime = _productionTimeLimit - _nbNodeSubdivisionRequests * _totalNodesProductionTime / _totalProducedNodes;
425 // Limit the number of request to fit in the remaining time
426 nbBricks = min(
427 nbBricks,
428 static_cast< uint >( remainingTime * _totalProducedBricks / _totalBrickProductionTime ) );
429 }
430
431 CUDAPM_START_EVENT( producer_bricks );
432 if ( nbBricks > 0 )
433 {
434 _nbBrickLoadRequests = manageDataProduction( nbBricks );
435 }
436 CUDAPM_STOP_EVENT( producer_bricks );
437#else
438 if ( nbRequests > 0 ) {
439 produceData( nbRequests );
440 }
441#endif
442 //}
443
444 // Tree data structure monitoring
445 if ( _hasTreeDataStructureMonitoring )
446 {
447 //if ( _nbNodeSubdivisionRequests > 0 )
448 //{
449 // TEST
450 dim3 blockSize( 128, 1, 1 );
451 dim3 gridSize( ( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z ) / 128 + 1, 1, 1 );
452 GVKernel_TrackLeafNodes< typename TDataStructure::VolTreeKernelType ><<< gridSize, blockSize >>>( _dataStructure->volumeTreeKernel, _nodesCacheManager->_pageTable->getKernel()/*page table*/, ( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z )/*nb nodes*/, _dataStructure->getMaxDepth()/*max depth*/, thrust::raw_pointer_cast( &( *this->_leafNodes )[ 0 ] ), thrust::raw_pointer_cast( &( *this->_emptyNodeVolume )[ 0 ] ) );
453 _nbLeafNodes = thrust::reduce( (*_leafNodes).begin(), (*_leafNodes).end(), static_cast< unsigned int >( 0 ), thrust::plus< unsigned int >() );
454 const float _emptyVolume = thrust::reduce( (*_emptyNodeVolume).begin(), (*_emptyNodeVolume).end(), static_cast< float >( 0.f ), thrust::plus< float >() );
455 //std::cout << "------------------------------------------------" << _nbLeafNodes << std::endl;
456 //std::cout << "Volume of empty nodes : " << ( _emptyVolume * 100.f ) << std::endl;
457 //std::cout << "------------------------------------------------" << _nbLeafNodes << std::endl;
458 //std::cout << "TOTAL number of leaf nodes : " << _nbLeafNodes << std::endl;
459 GVKernel_TrackNodes< typename TDataStructure::VolTreeKernelType ><<< gridSize, blockSize >>>( _dataStructure->volumeTreeKernel, _nodesCacheManager->_pageTable->getKernel()/*page table*/, ( _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z )/*nb nodes*/, _dataStructure->getMaxDepth()/*max depth*/, thrust::raw_pointer_cast( &( *this->_leafNodes )[ 0 ] ), thrust::raw_pointer_cast( &( *this->_emptyNodeVolume )[ 0 ] ) );
460 _nbNodes = thrust::reduce( (*_leafNodes).begin(), (*_leafNodes).end(), static_cast< unsigned int >( 0 ), thrust::plus< unsigned int >() );
461 //std::cout << "TOTAL number of nodes : " << _nbNodes << std::endl;
462 //}
463 if ( _nbBrickLoadRequests > 0 )
464 {
465 //...
466 }
467 }
468
469 if ( _lastProductionTimed && nbRequests > 0 )
470 {
471 cudaEventRecord( _stopProductionBricks );
472 }
473
474 return nbRequests;
475}

481void GsDataProductionManager< TDataStructure, TPriorityPoliciesManager >
482::clearCache()
483{
484 // Launch Kernel
485 dim3 blockSize( 32, 1, 1 ); // BEWARE : only works if nodetileSize < 32
486 dim3 gridSize( 1, 1, 1 );
487 // This clears node pool child and brick 1st nodetile after root node
488 GsKernel_ClearVolTreeRoot<<< gridSize, blockSize >>>( _dataStructure->volumeTreeKernel, NodeTileRes::getNumElements() );
489 GV_CHECK_CUDA_ERROR( "GsKernel_ClearVolTreeRoot" );
490 // TODO:
491 // - this is not good for Multi-Objects => it requires a modified or dedicated kernel
492 // ...
493
494 // Clear buffers holding mask of used and non-used elements
495 CUDAPM_START_EVENT( gpucachemgr_clear_fillML )
496 _dUnusedElementMasks->fillAsync( 0 );
497 _dUsedElementMasks->fill( 0 );
498 CUDAPM_STOP_EVENT( gpucachemgr_clear_fillML )
499
500 // Reset nodes cache manager
501 _nodesCacheManager->clearCache();
502//#ifndef GS_USE_MULTI_OBJECTS
503// _nodesCacheManager->_totalNumLoads = 2;
504// _nodesCacheManager->_lastNumLoads = 1;
505//#else
506// _nodesCacheManager->_totalNumLoads = 2 + 1;
507// _nodesCacheManager->_lastNumLoads = 1 + 1;
508//#endif
509 _nodesCacheManager->_totalNumLoads = 1/*protect first element*/ + _nbObjects;

511
512 // Reset bricks cache manager
513 _bricksCacheManager->clearCache();
514 _bricksCacheManager->_totalNumLoads = 0;
515 _bricksCacheManager->_lastNumLoads = 0;
516}

635 *
636 * @return The number of elements in the requests list
637 ******************************************************************************/
638template< typename TDataStructure, typename TPriorityPoliciesManager >
639uint GsDataProductionManager< TDataStructure, TPriorityPoliciesManager >
640::manageUpdates()
641{
642 uint nbElements = 0;
643
644 uint totalNbElements = _nodePoolRes.x * _nodePoolRes.y * _nodePoolRes.z;
645
646 // Optimisation test for case where the cache is not full
647 if ( _nodesCacheManager->_totalNumLoads < _nodesCacheManager->getNumElements() )
648 {
649 totalNbElements = ( _nodesCacheManager->_totalNumLoads ) * NodeTileRes::getNumElements();
650
651 // LOG
652 //std::cout << "totalNbElements : " << totalNbElements << "/" << nodePoolRes.x * nodePoolRes.y * nodePoolRes.z << "\n";
653 }
654
655 CUDAPM_START_EVENT( dataProduction_manageRequests_elemsReduction );
656
657 // Fill the buffer used to store node addresses updates with subdivision or load requests
658
659 // Retrieve masks of valid and unvalid requests
660 retrieveRequestMasks( totalNbElements );
661
662 // Retrieve list of valid requests
663 retrieveValidRequests( totalNbElements );
664
665#ifdef GS_USE_MULTI_OBJECTS
666 const bool result = GsCompute::GsDataParallelPrimitives::get().compact(
667 /* OUT : compacted output */_objectIDBufferCompactList->getPointer( 0 ),
668 /* OUT : number of elements valid flags in the d_isValid input array */_d_nbValidRequests,
669 /* input to compact */_objectIDBuffer->getPointer(),
670 /* which elements in input are valid */_d_validRequestMasks->getPointer(),
671 /* nb of elements in input */totalNbElements );
672 GV_CHECK_CUDA_ERROR( "KERNEL manageUpdates::cudppCompact objectIDs" );
673 assert( result );
674#endif
675
676 // If priority management is required, first retrieve the associated list of valid elements from the priority buffer
677 if ( _hasProductionPriority && TPriorityPoliciesManager::usePriority() )
678 {
679 // Retrieve list of valid request priorities
680 retrieveValidRequestPriorities( totalNbElements );
681 }
682
683#ifndef GS_USE_OPTIMIZED_NON_BLOCKING_ASYNCHRONOUS_CALLS_PIPELINE_DATAPRODUCTIONMANAGER
684 // Retrieve number of valid requests
685 nbElements = retrieveNbValidRequests();
686#endif
687
688 // If priority management is required, then sort the valid elements from the priority buffer
689 if ( _hasProductionPriority && TPriorityPoliciesManager::usePriority() )
690 {
691 // Sort requests using priorities
692 sortRequestPriorities( nbElements );
693 }
694
695 CUDAPM_STOP_EVENT( dataProduction_manageRequests_elemsReduction );
696
697 // Return number of valid requests
698 return nbElements;
