118 // Note on untransformed and transformed objects with the original and inverse transformed rays.
119 //
120 // Given a point P on an object, applying a model transformation T on it involves modifications.
121 //
122 // Given a starting point "O" on a ray of direction "D" (unit vector), a point at distance "t" on this ray
123 // is represented by in World space by :
124 // P' = O + t D
125 // And the corresponding point in Object space is :
126 // P = O' + t D'
127 // where O' = T-1 O and D' = T-1 D
128
129 // Calculate eye ray in tree space
130 //
131 // Apply the inverse set of transformations to the ray to produce an "inverse transformed ray"
132 float3 rayDir = k_renderViewContext.viewPlaneDirTP
133 + k_renderViewContext.viewPlaneXAxisTP * static_cast< float >( pixelCoords.x )
134 + k_renderViewContext.viewPlaneYAxisTP * static_cast< float >( pixelCoords.y );
135
136 // Ray start
137 smRayStart = k_renderViewContext.viewCenterTP;
138
139 // Not sure to normalize
140 // - traditionaly, in ray-tracing, for the object space, we don't have to normalize the ray
141 rayDir = normalize( rayDir );
142
143 // Intersect the inverse transformed ray with the untransformed object
144 // - a BBox in [ 0.0; 1.0 ] x [ 0.0; 1.0 ] x [ 0.0; 1.0 ]
145 float boxInterMin = 0.0f;
146 float boxInterMax = 10000.0f;
147 int hit = intersectBox( smRayStart, rayDir, make_float3( 0.001f ), make_float3( 0.999f ), boxInterMin, boxInterMax );
148 bool masked = ! ( hit && ( boxInterMax > 0.0f ) );

150 // Set closest hit point
151 boxInterMin = maxcc( boxInterMin, k_renderViewContext.frustumNear ); // TO DO : attention, c'est faux => frustumNear est en "espace camera" !!

154 // Set farthest hit point

157 {
158 // Retrieve the view-space depth from the depth buffer. Only works if w was 1.0f.

161 // Compute z in NDC space

163 // True equation is :
164 // float zNDC = ( 2.f * frameDepth - k_renderViewContext.depthRangeFar - k_renderViewContext.depthRangeNear ) / ( k_renderViewContext.depthRangeFar - k_renderViewContext.depthRangeNear ) - 1.f;
165 // but a simplication occurs if glDepthRange values are ( 0.0; 1.0 )
166 // float zNDC = 2.0f * frameDepth - 1.0f;
167 const float zNDC = 2.0f * frameDepth - 1.0f;
168
169 // Compute z in Eye space
170 const float zEye = k_renderViewContext.frustumD / ( -zNDC - k_renderViewContext.frustumC );
171
172 // Take minimum value between input depth and bbox output
173 tMax = mincc( -zEye, boxInterMax );
174 }
175
176 // Discard special cases
177 if ( t == 0.0f || t >= tMax )
178 {
179 masked = true;
180 }

182 // If intersection
183 if ( ! masked )
184 {
185 // Read color from the input color buffer
186 uchar4 frameColor = getInputColor( pixelCoords );
187
188 // Launch N3-tree traversal and rendering
189 CUDAPM_KERNEL_START_EVENT( pixelCoords, 1 );
190 GsRendererKernel::render< TFastUpdateMode, TPriorityOnBrick >( pVolumeTree, sampleShader, pCache, pixelCoords, smRayStart, rayDir, tMax, t );
191 CUDAPM_KERNEL_STOP_EVENT( pixelCoords, 1 );
192
193 // Retrieve the accumulated color along the ray
194 const float4 accCol = sampleShader.getColor();
195
196 // Convert color from uchar [ 0 ; 255 ] to float [ 0.0 ; 1.0 ]
197 float4 scenePixelColorF = make_float4( (float)frameColor.x / 255.0f, (float)frameColor.y / 255.0f, (float)frameColor.z / 255.0f, (float)frameColor.w / 255.0f );

199 // TO DO
200 // - let OpenGL do that operation with blending ?

202 // Blend colors (ray and framebuffer)
203 float4 pixelColorF = accCol + scenePixelColorF * ( 1.0f - accCol.w );

205 // Clamp color to be within the interval [+0.0, 1.0]
206 pixelColorF.x = __saturatef( pixelColorF.x );
207 pixelColorF.y = __saturatef( pixelColorF.y );
208 pixelColorF.z = __saturatef( pixelColorF.z );
209 //pixelColorF.w = 1.0f; // <== why 1.0f and not __saturatef( pixelColorF.w ) ? // Pour éviter une opération OpenGL de ROP ? Car ça a été penser pour rendre en dernier au départ ?
210 pixelColorF.w = __saturatef( pixelColorF.w );
211 //pixelColorF.w = __saturatef( pixelColorF.w );
212
213 // Convert color from float [ 0.0 ; 1.0 ] to uchar [ 0 ; 255 ]
214 frameColor = make_uchar4( (uchar)( pixelColorF.x * 255.0f ), (uchar)( pixelColorF.y * 255.0f ), (uchar)( pixelColorF.z * 255.0f ), (uchar)( pixelColorF.w * 255.0f ) );
215
216 // Project the depth and check against the current one
217 float pixDepth = 1.0f;
218 if ( accCol.w > cOpacityDepthThreshold )
219 {
220 const float VP = -fabsf( t * rayDir.z );
221 //http://www.opengl.org/discussion_boards/ubbthreads.php?ubb=showflat&Number=234519&page=2
222 const float clipZ = ( VP * k_renderViewContext.frustumC + k_renderViewContext.frustumD ) / -VP;
223
224 //pixDepth = clamp( ( clipZ + 1.0f ) / 2.0f, 0.0f, 1.0f ); // TO DO : use __saturatef instead !! =====> ( [ x 0.5f ] instead ) ??
225 pixDepth = __saturatef( clipZ * 0.5f + 0.5f );
226
227 //const float zNDC = - ( k_renderViewContext.frustumC + k_renderViewContext.frustumD / ( -t ) );
228 //const float zWindow = zNDC * 0.5f + * 0.5f;
229 //pixDepth = zWindow;
230 }
231 frameDepth = min( frameDepth, pixDepth );
232
233 // Write color in color buffer
234 setOutputColor( pixelCoords, frameColor );
235
236 // Write depth in depth buffer
237 setOutputDepth( pixelCoords, frameDepth );
238 }
239 } // !outOfFrame
240
241 CUDAPM_KERNEL_STOP_EVENT( pixelCoords, 0 );
