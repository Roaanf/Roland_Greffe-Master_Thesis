56/******************************************************************************
57 * DEVICE function
58 *
59 * This is the function where shading is done (ray marching along ray and data sampling).
60 * Shading is done with cone-tracing (LOD is selected by comparing cone aperture versus voxel size).
61 *
62 * @param pVolumeTree data structure
63 * @param pSampleShader shader
64 * @param pGpuCache cache (not used for the moment...)
65 * @param pRayStartTree camera position in Tree coordinate system
66 * @param pRayDirTree ray direction in Tree coordinate system
67 * @param pTTree the distance from the eye to current position along the ray
68 * @param pRayLengthInNodeTree the distance along the ray from start to end of the brick, according to ray direction
69 * @param pBrickSampler The object in charge of sampling data (i.e. texture fetches)
70 * @param pModifInfoWriten (not used for the moment...)
71 *
72 * @return the distance where ray-marching has stopped
73 ******************************************************************************/
74template< bool TFastUpdateMode, bool TPriorityOnBrick, class TVolumeTreeKernelType, class TSampleShaderType, class TGPUCacheType >
75__device__
76float GsBrickVisitorKernel
77::visit( TVolumeTreeKernelType& pVolumeTree, TSampleShaderType& pSampleShader,
78 TGPUCacheType& pGpuCache, const float3 pRayStartTree, const float3 pRayDirTree, const float pTTree,
79 const float pRayLengthInNodeTree, GsSamplerKernel< TVolumeTreeKernelType >& pBrickSampler, bool& pModifInfoWriten )
80{
81 // Current position in tree space
82 float3 samplePosTree = pRayStartTree + pTTree * pRayDirTree;
83
84 // Local distance
85 float dt = 0.0f;
86
87 // Step
88 float rayStep = 0.0f;
89
90 // Traverse the brick
91 while ( dt <= pRayLengthInNodeTree && !pSampleShader.stopCriterion( samplePosTree ) )
92 {
93 // Update global distance
94 float fullT = pTTree + dt;
95
96 // Get the cone aperture at the given distance
97 float coneAperture = pSampleShader.getConeAperture( fullT );
98
99 // Update sampler mipmap parameters
100 if ( ! pBrickSampler.updateMipMapParameters( coneAperture ) )
101 {
102 break;
103 }
104
105 // Move sampler position
106 pBrickSampler.moveSampleOffsetInNodeTree( rayStep * pRayDirTree );
107
108 // Update position
109 samplePosTree = pRayStartTree + fullT * pRayDirTree;
110
111 // Compute next step
112 //
113 // TO DO : check if coneAperture, based on radial distance to camera, could not generate spherical pattern
114 rayStep = max( coneAperture, pBrickSampler._nodeSizeTree * ( 0.66f / static_cast< float>( TVolumeTreeKernelType::BrickResolution::x ) ) );
115
116 // Shading (+ adaptative step)
117 pSampleShader.run( pBrickSampler, samplePosTree, pRayDirTree, rayStep, coneAperture );
118
119 // Update local distance
120 dt += rayStep;
121 }
122
123 return dt;
124}
