54/******************************************************************************
55 * KERNEL : CreateLocalizationLists
56 *
57 * Extract localization informations associated with a list of given elements.
58 *
59 * Special version for gpuProducerLoadCache that transform loccodes to usable ones (warning, loose one bit of depth !)
60 *
61 * @param pNbElements number of elements to process
62 * @param pLoadAddressList List of input node addresses
63 * @param pLocCodeList List of localization code coming from the main page table of the data structure and referenced by cache managers (nodes and bricks)
64 * @param pLocDepthList List of localization depth coming from the main page table of the data structure and referenced by cache managers (nodes and bricks)
65 * @param pResLocCodeList Resulting output localization code list
66 * @param pResLocDepthList Resulting output localization depth list
67 ******************************************************************************/
68template< class NodeTileRes >
69__global__
70void CreateLocalizationLists( const uint pNbElements, const uint* pLoadAddressList,
71 const GvCore::GsLocalizationInfo::CodeType* pLocCodeList, const GvCore::GsLocalizationInfo::DepthType* pLocDepthList,
72 GvCore::GsLocalizationInfo::CodeType* pResLocCodeList, GvCore::GsLocalizationInfo::DepthType* pResLocDepthList )
73{
74 // Retrieve global indexes
75 const uint lineSize = __uimul( blockDim.x, gridDim.x );
76 const uint elem = threadIdx.x + __uimul( blockIdx.x, blockDim.x ) + __uimul( blockIdx.y, lineSize );
77
78 // Check bounds
79 if ( elem < pNbElements )
80 {
81 // Retrieve node address and its localisation info
82 // along with new element address
83
84 // Retrieve node address
85 const uint nodeAddressEnc = pLoadAddressList[ elem ];
86 const uint3 nodeAddress = GvStructure::GsNode::unpackNodeAddress( nodeAddressEnc );
87
88 // Retrieve its "node tile" address
89 const uint nodeTileAddress = nodeAddress.x / NodeTileRes::getNumElements();
90
91 // Retrieve its "localization info"
92 const GvCore::GsLocalizationInfo::CodeType *tileLocCodeEnc = &pLocCodeList[ nodeTileAddress ];
93
94 // Linear offset of the node in the node tile
95 const uint linearOffset = nodeAddress.x - ( nodeTileAddress * NodeTileRes::getNumElements() );
96 // 3D offset of the node in the node tile
97 const uint3 nodeOffset = NodeTileRes::toFloat3( linearOffset );
98
99 const GvCore::GsLocalizationInfo::CodeType nodeLocCodeEnc = tileLocCodeEnc->addLevel< NodeTileRes >( nodeOffset );
100
101 // Write localization info
102 pResLocDepthList[ elem ] = pLocDepthList[ nodeTileAddress ];
103 pResLocCodeList[ elem ] = nodeLocCodeEnc;
104 }
105}

154/******************************************************************************
155 * Return the localization info of a node in the node pool
156 *
157 * @param nodeAddress Address of the node in the node pool
158 *
159 * @return The localization info of the node
160 ******************************************************************************/
161template< typename NodeTileRes, typename ElementRes, typename AddressType, typename KernelArrayType, typename LocCodeArrayType, typename LocDepthArrayType >
162__device__
163__forceinline__ GsLocalizationInfo PageTableNodesKernel< NodeTileRes, ElementRes, AddressType, KernelArrayType, LocCodeArrayType, LocDepthArrayType >
164::getLocalizationInfoImpl( uint nodeAddress ) const
165{
166 // Compute the address of the current node tile (and its offset in the node tile)
167 uint nodeTileIndex = nodeAddress / NodeTileRes::getNumElements();
168 uint nodeTileAddress = nodeTileIndex * NodeTileRes::getNumElements();
169 uint nodeTileOffset = nodeAddress - nodeTileAddress;
170
171 // Compute the node offset (in 3D, in the node tile)
172 uint3 nodeOffset = NodeTileRes::toFloat3( nodeTileOffset );
173
174 // Fetch associated localization infos
175 GsLocalizationInfo::CodeType parentLocCode = locCodeArray.get( nodeTileIndex );
176 GsLocalizationInfo::DepthType parentLocDepth = locDepthArray.get( nodeTileIndex );
177
178 // Localization info initialization
179 GsLocalizationInfo locInfo;
180 locInfo.locCode = parentLocCode.addLevel< NodeTileRes >( nodeOffset );
181 locInfo.locDepth = parentLocDepth;
182
183 return locInfo;
184}

196::setPointerImpl( uint elemAddress, ElemAddressType elemPointer, uint flags )
197{
198 ElemPackedAddressType packedChildAddress = childArray.get( elemAddress );
199 ElemPackedAddressType packedAddress = AddressType::packAddress( elemPointer );
200
201 // Update node tile's pointer
202 childArray.set( elemAddress,
203 /*remove the fact that a node could be terminal*/( packedChildAddress & 0x40000000 ) | /*add children's nodetile address*/( packedAddress & 0x3FFFFFFF ) );
204
205 // Compute the address of the current node tile
206 uint nodeTileIndex = elemAddress / NodeTileRes::getNumElements();
207 uint nodeTileAddress = nodeTileIndex * NodeTileRes::getNumElements();
208 uint nodeTileOffset = elemAddress - nodeTileAddress;
209
210 // Compute the node offset
211 uint3 nodeOffset = NodeTileRes::toFloat3( nodeTileOffset );
212
213 // Fetch associated localization infos
214 GsLocalizationInfo::CodeType parentLocCode = locCodeArray.get( nodeTileIndex );
215 GsLocalizationInfo::DepthType parentLocDepth = locDepthArray.get( nodeTileIndex );
216
217 // Compute the address of the new node tile
218 uint newNodeTileIndex = elemPointer.x / ElementRes::getNumElements();
219 //uint newNodeTileAddress = newNodeTileIndex * ElementRes::getNumElements(); // --> semble ne pas être utilisé ?
220
221 // Update associated localization infos
222 GsLocalizationInfo::CodeType newLocCode = parentLocCode.addLevel< NodeTileRes >( nodeOffset );
223 GsLocalizationInfo::DepthType newLocDepth = parentLocDepth.addLevel();
224
225 locCodeArray.set( newNodeTileIndex, newLocCode );
226 locDepthArray.set( newNodeTileIndex, newLocDepth );
