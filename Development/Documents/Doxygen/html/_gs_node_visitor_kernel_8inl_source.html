56/******************************************************************************
57 * Descent in data structure (in general octree) until max depth is reach or current traversed node has no subnodes,
58 * or cone aperture is greater than voxel size.
59 *
60 * @param pVolumeTree the data structure
61 * @param pGpuCache the cache
62 * @param node a node that user has to provide. It will be filled with the final node of the descent
63 * @param pSamplePosTree A given position in tree
64 * @param pConeAperture A given cone aperture
65 * @param pNodeSizeTree the returned node size
66 * @param pSampleOffsetInNodeTree the returned sample offset in node tree
67 * @param pBrickSampler The sampler object used to sample data in the data structure, it will be initialized after the descent
68 * @param pRequestEmitted a returned flag to tell wheter or not a request has been emitted during descent
69 ******************************************************************************/
70template< bool priorityOnBrick, class TVolTreeKernelType, class GPUCacheType >
71__device__
72__forceinline__ void GsNodeVisitorKernel
73::visit( TVolTreeKernelType& pVolumeTree, GPUCacheType& pGpuCache, GvStructure::GsNode& pNode,
74 const float3 pSamplePosTree, const float pConeAperture, float& pNodeSizeTree, float3& pSampleOffsetInNodeTree,
75 GsSamplerKernel< TVolTreeKernelType >& pBrickSampler, bool& pRequestEmitted )
76{
77 // Useful variables initialization
78 uint nodeDepth = 0;
79 float3 nodePosTree = make_float3( 0.0f );
80 pNodeSizeTree = static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes );
81 float nodeSizeTreeInv = 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes );
82 float voxelSizeTree = pNodeSizeTree / static_cast< float >( TVolTreeKernelType::BrickResolution::maxRes );
83
84 uint brickChildAddressEnc = 0;
85 uint brickParentAddressEnc = 0;
86
87 float3 brickChildNormalizedOffset = make_float3( 0.0f );
88 float brickChildNormalizedScale = 1.0f;
89
90 // Initialize the address of the first node in the "node pool".
91 // While traversing the data structure, this address will be
92 // updated to the one associated to the current traversed node.
93 // It will be used to fetch info of the node stored in then "node pool".
94 uint nodeTileAddress = pVolumeTree._rootAddress;
95
96 // Traverse the data structure from root node
97 // until a descent criterion is not fulfilled anymore.
98 bool descentSizeCriteria;
99 do
100 {
101 // [ 1 ] - Update size parameters
102 pNodeSizeTree *= 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // current node size
103 voxelSizeTree *= 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // current voxel size
104 nodeSizeTreeInv *= static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // current node resolution (nb nodes in a dimension)
105
106 // [ 2 ] - Update node info
107 //
108 // The goal is to fetch info of the current traversed node from the "node pool"
109 uint3 nodeChildCoordinates = make_uint3( nodeSizeTreeInv * ( pSamplePosTree - nodePosTree ) );
110 uint nodeChildAddressOffset = TVolTreeKernelType::NodeResolution::toFloat1( nodeChildCoordinates );// & nodeChildAddressMask;
111 uint nodeAddress = nodeTileAddress + nodeChildAddressOffset;
112 nodePosTree = nodePosTree + pNodeSizeTree * make_float3( nodeChildCoordinates );
113 // Try to retrieve node from the node pool given its address
114 //pVolumeTree.fetchNode( pNode, nodeTileAddress, nodeChildAddressOffset );
115 pVolumeTree.fetchNode( pNode, nodeAddress );
116
117 // Update brick info
118 if ( brickChildAddressEnc )
119 {
120 brickParentAddressEnc = brickChildAddressEnc;
121 brickChildNormalizedScale = 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // 0.5f;
122 brickChildNormalizedOffset = brickChildNormalizedScale * make_float3( nodeChildCoordinates );
123 }
124 else
125 {
126 brickChildNormalizedScale *= 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // 0.5f;
127 brickChildNormalizedOffset += brickChildNormalizedScale * make_float3( nodeChildCoordinates );
128 }
129 brickChildAddressEnc = pNode.hasBrick() ? pNode.getBrickAddressEncoded() : 0;
130
131 // Update descent condition
132 descentSizeCriteria = ( voxelSizeTree > pConeAperture ) && ( nodeDepth < k_maxVolTreeDepth );
133
134 // Update octree depth
135 ++nodeDepth;
136
137 // ---- Flag used data (the traversed one) ----
138
139 // Set current node as "used"
140 pGpuCache._nodeCacheManager.setElementUsage( nodeTileAddress );
141
142 // Set current brick as "used"
143 if ( pNode.hasBrick() )
144 {
145 pGpuCache._brickCacheManager.setElementUsage( pNode.getBrickAddress() );
146 }
147
148 // ---- Emit requests if needed (node subdivision or brick loading/producing) ----
149
150 // Process requests based on traversal strategy (priority on bricks or nodes)
151 GvCore::GsPriorityParameters priorityParams;
152 priorityParams._coneAperture = pConeAperture;
153 priorityParams._nodeSize = pNodeSizeTree;
154 priorityParams._nodeDepth = nodeDepth;
155
156 if ( priorityOnBrick )
157 {
158 // Low resolution first
159 if ( ( pNode.isBrick() && !pNode.hasBrick() ) || !( pNode.isInitializated() ) )
160 {
161 pGpuCache.loadRequest( nodeAddress, priorityParams );
162 pRequestEmitted = true;
163 }
164 else if ( !pNode.hasSubNodes() && descentSizeCriteria && !pNode.isTerminal() )
165 {
166 pGpuCache.subDivRequest( nodeAddress, priorityParams );
167 pRequestEmitted = true;
168 }
169 }
170 else
171 { // High resolution immediately
172 if ( descentSizeCriteria && !pNode.isTerminal() )
173 {
174 if ( ! pNode.hasSubNodes() )
175 {
176 pGpuCache.subDivRequest( nodeAddress, priorityParams );
177 pRequestEmitted = true;
178 }
179 }
180 else if ( ( pNode.isBrick() && !pNode.hasBrick() ) || !( pNode.isInitializated() ) )
181 {
182 pGpuCache.loadRequest( nodeAddress, priorityParams );
183 pRequestEmitted = true;
184 }
185 }
186
187 nodeTileAddress = pNode.getChildAddress().x;
188 }
189 while ( descentSizeCriteria && pNode.hasSubNodes() ); // END of the data structure traversal
190
191 // Compute sample offset in node tree
192 pSampleOffsetInNodeTree = pSamplePosTree - nodePosTree;
193
194 // Update brickSampler properties
195 //
196 // The idea is to store useful variables that will ease the rendering process of this node :
197 // - brickSampler is just a wrapper on the datapool to be able to fetch data inside
198 // - given the previously found node, we store its associated brick address in cache to be able to fetch data in the datapool
199 // - we can also store the brick address of the parent node to do linear interpolation of the two level of resolution
200 // - for all of this, we store the bottom left position in cache of the associated bricks (note : brick address is a voxel index in the cache)
201 if ( pNode.isBrick() )
202 {
203 pBrickSampler._nodeSizeTree = pNodeSizeTree;
204 pBrickSampler._sampleOffsetInNodeTree = pSampleOffsetInNodeTree;
205 pBrickSampler._scaleTree2BrickPool = pVolumeTree.brickSizeInCacheNormalized.x / pBrickSampler._nodeSizeTree;
206
207 pBrickSampler._brickParentPosInPool = pVolumeTree.brickCacheResINV * make_float3( GvStructure::GsNode::unpackBrickAddress( brickParentAddressEnc ) )
208 + brickChildNormalizedOffset * pVolumeTree.brickSizeInCacheNormalized.x;
209
210 if ( brickChildAddressEnc )
211 {
212 // Should be mipmapping here, betwwen level with the parent
213
214 //pBrickSampler.mipMapOn = true; // "true" is not sufficient : when no parent, program is very slow
215 pBrickSampler._mipMapOn = ( brickParentAddressEnc == 0 ) ? false : true;
216
217 pBrickSampler._brickChildPosInPool = make_float3( GvStructure::GsNode::unpackBrickAddress( brickChildAddressEnc ) ) * pVolumeTree.brickCacheResINV;
218 }
219 else
220 {
221 // No mipmapping here
222
223 pBrickSampler._mipMapOn = false;
224 pBrickSampler._brickChildPosInPool = pBrickSampler._brickParentPosInPool;
225 pBrickSampler._scaleTree2BrickPool *= brickChildNormalizedScale;
226 }
227 }
228}

230/******************************************************************************
231 * Descent in volume tree until max depth is reach or current traversed node has no subnodes.
232 * Perform a descent in a volume tree from a starting node tile address, until a max depth
233 * Given a 3D sample position,
234 *
235 * @param pVolumeTree The volume tree on which descent in done
236 * @param pMaxDepth Max depth of the descent
237 * @param pSamplePos 3D sample position
238 * @param pNodeTileAddress ...
239 * @param pNode ...
240 * @param pNodeSize ...
241 * @param pNodePos ...
242 * @param pNodeDepth ...
243 * @param pBrickAddressEnc ...
244 * @param pBrickPos ...
245 * @param pBrickScale ...
246 ******************************************************************************/
247template< class VolumeTreeKernelType >
248__device__
249__forceinline__ void GsNodeVisitorKernel
250::visit( VolumeTreeKernelType& pVolumeTree, uint pMaxDepth, float3 pSamplePos,
251 uint pNodeTileAddress, GvStructure::GsNode& pNode, float& pNodeSize, float3& pNodePos, uint& pNodeDepth,
252 uint& pBrickAddressEnc, float3& pBrickPos, float& pBrickScale )
253{
255
256 float nodeSizeInv = 1.0f;
257
258 // WARNING uint nodeAddress;
259 pBrickAddressEnc = 0;
260
261 // Descent in volume tree until max depth is reach or current traversed node has no subnodes
262 int i = 0;
263 do
264 {
265 pNodeSize *= 1.0f / (float)VolumeTreeKernelType::NodeResolution::maxRes;
266 nodeSizeInv *= (float)VolumeTreeKernelType::NodeResolution::maxRes;
267
268 // Retrieve current voxel position
269 uint3 curVoxel = make_uint3( nodeSizeInv * ( pSamplePos - pNodePos ) );
270 uint curVoxelLinear = VolumeTreeKernelType::NodeResolution::toFloat1( curVoxel );
271
272 float3 nodeposloc = make_float3( curVoxel ) * pNodeSize;
273 pNodePos = pNodePos + nodeposloc;
274
275 // Retrieve pNode info (child and data addresses) from pNodeTileAddress address and curVoxelLinear offset
276 pVolumeTree.fetchNode( pNode, pNodeTileAddress, curVoxelLinear );
277
278 if ( pNode.hasBrick() )
279 {
280 pBrickAddressEnc = pNode.getBrickAddressEncoded();
281 pBrickPos = make_float3( 0.0f );
282 pBrickScale = 1.0f;
283 }
284 else
285 {
286 pBrickScale = pBrickScale * 0.5f;
287 pBrickPos += make_float3( curVoxel ) * pBrickScale;
288 }
289
290 pNodeTileAddress = pNode.getChildAddress().x;
291 i++;
292 }
293 while ( ( i < pMaxDepth ) && pNode.hasSubNodes() );
294
295 pNodeDepth = i;
296
297 //i -= 1; // <== TODO : don't seem to be used anymore, remove it
298}

300/******************************************************************************

304 * @param pVolumeTree the data structure

316template< class TVolTreeKernelType >
317__device__
318__forceinline__ uint GsNodeVisitorKernel
319::getNodeFather( TVolTreeKernelType& pVolumeTree, GvStructure::GsNode& pNode, const float3 pSamplePosTree, const uint pMaxNodeDepth )
320{
321 // Useful variables initialization
322 uint nodeDepth = 0;
323 float3 nodePosTree = make_float3( 0.0f );
324 float nodeSizeTree = static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes );
325 float nodeSizeTreeInv = 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes );
326
327 // Initialize the address of the first node in the "node pool".
328 // While traversing the data structure, this address will be
329 // updated to the one associated to the current traversed node.
330 // It will be used to fetch info of the node stored in the "node pool".
331 uint nodeTileAddress = pVolumeTree._rootAddress;
332
333 // Traverse the data structure from root node
334 // until a descent criterion is not fulfilled anymore.
335 do
336 {
337 // [ 1 ] - Update size parameters
338 nodeSizeTree *= 1.0f / static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // current node size
339 nodeSizeTreeInv *= static_cast< float >( TVolTreeKernelType::NodeResolution::maxRes ); // current node resolution (nb nodes in a dimension)
340
341 // [ 2 ] - Update node info
342 //
343 // The goal is to fetch info of the current traversed node from the "node pool"
344 uint3 nodeChildCoordinates = make_uint3( nodeSizeTreeInv * ( pSamplePosTree - nodePosTree ) );
345 uint nodeChildAddressOffset = TVolTreeKernelType::NodeResolution::toFloat1( nodeChildCoordinates );// & nodeChildAddressMask;
346 uint nodeAddress = nodeTileAddress + nodeChildAddressOffset;
347 nodePosTree = nodePosTree + nodeSizeTree * make_float3( nodeChildCoordinates );
348 // Try to retrieve node from the node pool given its address
349 //pVolumeTree.fetchNode( pNode, nodeTileAddress, nodeChildAddressOffset );
350 pVolumeTree.fetchNode( pNode, nodeAddress );
351
352 nodeTileAddress = pNode.getChildAddress().x;
353
354 // Update depth
355 nodeDepth++;
356 }
357 while ( ( nodeDepth <= pMaxNodeDepth ) && pNode.hasSubNodes() ); // END of the data structure traversal
358
359 return nodeDepth;
360}
