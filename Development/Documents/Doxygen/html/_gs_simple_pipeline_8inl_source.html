86/******************************************************************************
87 * Initialize
88 *
89 * @param pNodePoolMemorySize Node pool memory size
90 * @param pBrickPoolMemorySize Brick pool memory size
91 * @param pShader the shader
92 * @param pUseGraphicsLibraryInteroperability a flag telling whether or not to use GL interoperability mode
93 ******************************************************************************/
94template< typename TShaderType, typename TDataStructureType, typename TCacheType >
95inline void GsSimplePipeline< TShaderType, TDataStructureType, TCacheType >
96::initialize( size_t pNodePoolMemorySize, size_t pBrickPoolMemorySize, TShaderType* pShader, bool pUseGraphicsLibraryInteroperability )
97{
98 assert( pNodePoolMemorySize > 0 );
99 assert( pBrickPoolMemorySize > 0 );
100 assert( pShader != NULL );
101
102 // Print datatype info
103 printDataTypeInfo();
104
105 // Store shader
106 _shader = pShader;
107
108 // Store global memory size of the pools
109 _nodePoolMemorySize = pNodePoolMemorySize;
110 _brickPoolMemorySize = pBrickPoolMemorySize;
111
112 // Compute the resolution of the pools
113 uint3 nodePoolResolution;
114 uint3 brickPoolResolution;
115 computePoolResolution( nodePoolResolution, brickPoolResolution );
116
117 std::cout << "\nNode pool resolution : " << nodePoolResolution << std::endl;
118 std::cout << "Brick pool resolution : " << brickPoolResolution << std::endl;
119
120 // Retrieve the requested graphics library interoperability mode
121 const unsigned int useGraphicsLibraryInteroperability = pUseGraphicsLibraryInteroperability ? 1 : 0;
122
123 // Data structure
124 _dataStructure = new DataStructureType( nodePoolResolution, brickPoolResolution, useGraphicsLibraryInteroperability );
125 assert( _dataStructure != NULL );
126
127 // Cache
128 _cache = new CacheType( _dataStructure, nodePoolResolution, brickPoolResolution, useGraphicsLibraryInteroperability );
129 assert( _cache != NULL );
130}

168/******************************************************************************
169 * Launch the main GigaSpace flow sequence
170 ******************************************************************************/
171template< typename TShaderType, typename TDataStructureType, typename TCacheType >
172inline void GsSimplePipeline< TShaderType, TDataStructureType, TCacheType >
173::execute( const float4x4& pModelMatrix, const float4x4& pViewMatrix, const float4x4& pProjectionMatrix, const int4& pViewport )
174{
175 // Check if a "clear request" has been asked
176 if ( _clearRequested )
177 {
178 CUDAPM_START_EVENT( gpucache_clear );
179
180 // Clear the cache
181 _cache->clearCache();
182
183 // Bug [#16161] "Cache : not cleared as it should be"
184 //
185 // Without the following clear of the data structure node pool, artefacts should appear.
186 // - it is visible in the Slisesix and ProceduralTerrain demos.
187 //
188 // It seems that the brick addresses of the node pool need to be reset.
189 // Maybe it's a problem of "time stamp" and index of current frame (or time).
190 //
191 // @todo : study this problem
192 _dataStructure->clearVolTree();
193
194 CUDAPM_STOP_EVENT( gpucache_clear );
195
196 // Update "clear request" flag
197 _clearRequested = false;
198 }
199
200#ifndef GS_USE_MULTI_OBJECTS
201
202 // Map resources()
203 // - this function provides the synchronization guarantee that any graphics calls issued before cudaGraphicsMapResources() will complete before any subsequent CUDA work in stream begins.
204 _renderer->preRender( pModelMatrix, pViewMatrix, pProjectionMatrix, pViewport );
205
206 // [ 1 ] - Pre-render stage
207 _cache->preRenderPass();
208
209 // [ 2 ] - Rendering stage
210 _renderer->render( pModelMatrix, pViewMatrix, pProjectionMatrix, pViewport );
211
212 // Unmap resources()
213 _renderer->postRender( pModelMatrix, pViewMatrix, pProjectionMatrix, pViewport );
214
215#else
216
217 // [ 1 ] - Pre-render stage
218 //_cache->preRenderPass();
219
220 // [ 2 ] - Rendering stage
221 //for ( size_t i = 0; i < _renderers.size(); i++ )
222 //{
223 //}
224
225#endif
226
227 // [ 3 ] - Post-render stage (i.e. Data Production Management)
228 CUDAPM_START_EVENT( dataProduction_handleRequests );
229 if ( _dynamicUpdate )
230 {
231 _cache->_intraFramePass = false;
232
233 // Post render pass
234 // This is where requests are processed : produce or load data
235 _cache->handleRequests();
236 }
237 CUDAPM_STOP_EVENT( dataProduction_handleRequests );
238}

465/******************************************************************************
466 * Compute the resolution of the pools
467 *
468 * @param pNodePoolResolution Node pool resolution
469 * @param pBrickPoolResolution Brick pool resolution
470 ******************************************************************************/
471template< typename TShaderType, typename TDataStructureType, typename TCacheType >
472inline void GsSimplePipeline< TShaderType, TDataStructureType, TCacheType >
473::computePoolResolution( uint3& pNodePoolResolution, uint3& pBrickPoolResolution )
474{
475 assert( _nodePoolMemorySize != 0 );
476 assert( _brickPoolMemorySize != 0 );
477
478 // Compute the size of one element in the cache for nodes and bricks
479 size_t nodeTileMemorySize = NodeTileResolution::numElements * sizeof( GvStructure::GsNode );
480 size_t brickTileMemorySize = RealBrickTileResolution::numElements * GvCore::DataTotalChannelSize< DataTypeList >::value;
481
482 // Compute how many we can fit into the given memory size
483 size_t nodePoolNbElements = _nodePoolMemorySize / nodeTileMemorySize;
484 size_t brickPoolNbElements = _brickPoolMemorySize / brickTileMemorySize;
485
486 // Compute the resolution of the pools
487 pNodePoolResolution = make_uint3( static_cast< uint >( floorf( powf( static_cast< float >( nodePoolNbElements ), 1.0f / 3.0f ) ) ) ) * NodeTileResolution::get();
488 pBrickPoolResolution = make_uint3( static_cast< uint >( floorf( powf( static_cast< float >( brickPoolNbElements ), 1.0f / 3.0f ) ) ) ) * RealBrickTileResolution::get();
489}
