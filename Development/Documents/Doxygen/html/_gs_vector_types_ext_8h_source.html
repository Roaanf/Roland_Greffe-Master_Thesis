196inline unsigned short __float2half_rn_host(float f)
197{
198 unsigned int x = *((unsigned int *)&f);
199 unsigned int u = (x & 0x7fffffff), remainder, shift, lsb, lsb_s1, lsb_m1;
200 unsigned int sign, exponent, mantissa;
201
202 if (u > 0x7f800000)
203 return 0x7fff;
204
205 sign = ((x >> 16) & 0x8000);
206
207 if (u > 0x477fefff)
208 return sign | 0x7c00;
209
210 if (u < 0x33000001)
211 return sign | 0x0000;
212
213 exponent = ((u >> 23) & 0xff);
214 mantissa = (u & 0x7fffff);
215
216 if (exponent > 0x70) {
217 shift = 13;
218 exponent -= 0x70;
219 } else {
220 shift = 0x7e - exponent;
221 exponent = 0;
222 mantissa |= 0x800000;
223 }
224 lsb = (1 << shift);
225 lsb_s1 = (lsb >> 1);
226 lsb_m1 = (lsb - 1);
227
228 remainder = (mantissa & lsb_m1);
229 mantissa >>= shift;
230 if (remainder > lsb_s1 || (remainder == lsb_s1 && (mantissa & 0x1))) {
231 ++mantissa;
232 if (!(mantissa & 0x3ff)) {
233 ++exponent;
234 mantissa = 0;
235 }
236 }
237
238 return sign | (exponent << 10) | mantissa;
239}

924 float w2=w*w;

926 float y2=y*y;
927 float z2=z*z;

931 (src).x*2*(x*y-w*z) + (src).y*(w2-x2+y2-z2) + (src).z*2*(y*z+w*x),
