110 // Create texture containing the lookup table.
111 const unsigned int nElem = sizeof( permutationTable ) / sizeof( permutationTable[ 0 ] );
112 const unsigned int lookupTableNElem = nElem * nElem;
113
114 // Pre-compute table
115 uchar4* tmpTable = static_cast< uchar4* >( malloc( lookupTableNElem * sizeof( uchar4 ) ) );
116
117 for ( unsigned int y = 0; y < nElem; ++y )
118 {
119 unsigned int col = y * nElem;

144 gs_permutationTableTexture.addressMode[ 1 ] = cudaAddressModeWrap;
145 gs_permutationTableTexture.addressMode[ 2 ] = cudaAddressModeWrap;
146
147 free( tmpTable );
148 }

182 gs_gradientTexture.filterMode = cudaFilterModePoint;
183 gs_gradientTexture.addressMode[ 0 ] = cudaAddressModeWrap; // Wrap texture coordinates

185 gs_gradientTexture.addressMode[ 2 ] = cudaAddressModeWrap;
