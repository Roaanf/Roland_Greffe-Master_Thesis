169/******************************************************************************
170 * KERNEL GsKernel_ClearVolTreeRoot
171 *
172 * This clears node pool child and brick 1st nodetile after root node.
173 *
174 * @param pDataStructure data structure
175 * @param pRootAddress root node address from which to clear data
176 *
177 * TODO:
178 * - this is not good for Multi-Objects => it requires a modified or dedicated kernel
179 ******************************************************************************/
180template< typename VolTreeKernelType >
181__global__
182// __launch_bounds__( maxThreadsPerBlock, minBlocksPerMultiprocessor )
183void GsKernel_ClearVolTreeRoot( VolTreeKernelType pDataStructure, const uint pRootAddress )
184{
185 //uint lineSize = __uimul( blockDim.x, gridDim.x );
186 const uint elem = threadIdx.x + __uimul( blockIdx.x, blockDim.x );// + __uimul( blockIdx.y, lineSize );
187
188 if ( elem < VolTreeKernelType::NodeResolution::getNumElements() )
189 {
190 GvStructure::GsNode node;
191 node.childAddress = 0;
192 node.brickAddress = 0;
193#ifdef GS_USE_NODE_META_DATA
194 node._metaData = 0;
195#endif
196 pDataStructure.setNode( node, pRootAddress + elem );
197 }
198}

211{
212 uint elemNum = blockIdx.x;
213
214 if ( elemNum < numElems )
215 {
216 uint elemIndexEnc = lruElemAddressList[ elemNum ];
217 uint3 elemIndex = GvStructure::VolTreeBrickAddress::unpackAddress( elemIndexEnc );
218 uint3 elemAddress = elemIndex * ElementRes::get();
219
220 // FIXME: fixed border size !
221 uint3 brickAddress = elemAddress + make_uint3( 1 );
222 gpuCache._brickCacheManager.setElementUsage( brickAddress );
223 }
