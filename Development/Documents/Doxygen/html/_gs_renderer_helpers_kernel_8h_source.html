101 __device__
102 __forceinline__ int intersectBox(Ray r, const float3 boxmin, const float3 boxmax, float &tnear, float &tfar)
103 {
104 // compute intersection of ray with all six bbox planes
105 float3 invR = make_float3(1.0f) / r.dir;
106 float3 tbot = invR * (boxmin - r.start);
107 float3 ttop = invR * (boxmax - r.start);
108
109 // re-order intersections to find smallest and largest on each axis
110 float3 tmin = fminf(ttop, tbot);
111 float3 tmax = fmaxf(ttop, tbot);
112
113 // find the largest tmin and the smallest tmax
114 float largest_tmin = fmaxf(fmaxf(tmin.x, tmin.y), fmaxf(tmin.x, tmin.z));
115 float smallest_tmax = fminf(fminf(tmax.x, tmax.y), fminf(tmax.x, tmax.z));
116
117 tnear = largest_tmin;
118 tfar = smallest_tmax;
119
120 return smallest_tmax > largest_tmin;
121 }

174 /******************************************************************************
175 * Helper function that compute the mip-mapping coefficient
176 * to use during interpolation of two levels of resolution.
177 *
178 * @param pConeAperture cone aperture
179 * @param pNodeSize node size
180 ******************************************************************************/
181 template< typename TNodeRes, typename TBrickRes >
182 __device__
183 __forceinline__ float getMipMapInterpCoef( const float pConeAperture, const float pNodeSize )
184 {
185 // Compute ratio between node size and current cone aperture
186 float voxelSizeInv = static_cast< float >( TBrickRes::maxRes ) / pNodeSize;
187 float x = pConeAperture * voxelSizeInv;
188
189 // NOTE : log2( 1 / voxelSize ) = -log2( voxelSize ) ==> indicates the depth
190
191 // Handle different cases according to node resolution
192 if ( TNodeRes::x == 2 && TNodeRes::y == 2 && TNodeRes::z == 2 )
193 {
194 return __log2f( x );
195 }
196 else
197 {
198 // This case appears in the Menger Sponge tutorial,
199 // where 3x3 node tiles are used instead of octrees.
200 //
201 // TO DO : validate this with Fabrice
202 return __logf( x ) / __logf( static_cast< float >( TNodeRes::x ) );
203 }
204 }

206 /******************************************************************************
207 * Helper function to get ray length in node
208 *
209 * @param sampleOffsetInNodeTree sample offset in node
210 * @param nodeSizeTree size of node
211 * @param rayDirTree ray direction
212 *
213 * @return ...
214 ******************************************************************************/
215 __device__
216 __forceinline__ float getRayLengthInNode( const float3 sampleOffsetInNodeTree, const float nodeSizeTree, const float3 rayDirTree )
217 {
218 float3 directions = stepZero( rayDirTree ); // To precompute somewhere
219
220 float3 planes = directions * nodeSizeTree;
221
222 float3 distToBorder = ( planes - sampleOffsetInNodeTree ) / rayDirTree;
223
224 return fminf( distToBorder.x, fminf( distToBorder.y, distToBorder.z ) );
225 }

273 /******************************************************************************
274 * Get the color at given pixel from input color buffer
275 *
276 * @param pPixel pixel coordinates
277 *
278 * @return the pixel color
279 ******************************************************************************/
280 __device__
281 __forceinline__ uchar4 getInputColor( const uint2 pixelCoords )
282 {
283 uchar4 ret;
284
285 switch ( k_renderViewContext._graphicsResourceAccess[ GsGraphicsInteroperabiltyHandler::eColorInput ] )
286 {
287 case GsGraphicsResource::ePointer:
288 {
289 int offset = pixelCoords.x + pixelCoords.y * k_renderViewContext.frameSize.x;
290 ret = static_cast< uchar4* >( k_renderViewContext._graphicsResources[ GsGraphicsInteroperabiltyHandler::eColorInput ] )[ offset ];
291 break;
292 }
293 case GsGraphicsResource::eTexture:
294 {
295 ret = tex2D( _inputColorTexture, k_renderViewContext._inputColorTextureOffset + pixelCoords.x, pixelCoords.y );
296 break;
297 }
298 case GsGraphicsResource::eSurface:
299 {
300 // Note : cudaBoundaryModeTrap means that out-of-range accesses cause the kernel execution to fail.
301 // Possible values can be : cudaBoundaryModeTrap, cudaBoundaryModeClamp or cudaBoundaryModeZero.
302 ret = surf2Dread< uchar4 >( _colorSurface, pixelCoords.x * sizeof( uchar4 ), pixelCoords.y, cudaBoundaryModeTrap );
303 break;
304 }
305 default:
306 {
307 ret = k_renderViewContext._clearColor;
308 break;
309 }
310 }
311
312 return ret;
313 }

412 __device__
413 __forceinline__ void setOutputDepth( const uint2 pixelCoords, float depth )
414 {
415 // Encode depth to Z-buffer
416 uint tmpival = static_cast< uint >( floorf( depth * 16777215.0f ) );
417 tmpival = tmpival << 8;
418 float Zdepth = __int_as_float( tmpival );
419
420 // Write depth to Z-buffer

426 static_cast< float* >( k_renderViewContext._graphicsResources[ GsGraphicsInteroperabiltyHandler::eDepthOutput ] )[ offset ] = Zdepth;
427 }
428 break;
429
430 case GsGraphicsResource::eSurface:
431 // Note : cudaBoundaryModeTrap means that out-of-range accesses cause the kernel execution to fail.
432 // Possible values can be : cudaBoundaryModeTrap, cudaBoundaryModeClamp or cudaBoundaryModeZero.
433 surf2Dwrite( depth, _depthSurface, pixelCoords.x * sizeof( float ), pixelCoords.y, cudaBoundaryModeTrap );
434 break;
435
436 default:
437 break;

450 {
451 uint res;
452
453 input.x = (input.x | (input.x << 8)) & 0x00FF00FF;
454 input.x = (input.x | (input.x << 4)) & 0x0F0F0F0F;
455 input.x = (input.x | (input.x << 2)) & 0x33333333;
456 input.x = (input.x | (input.x << 1)) & 0x55555555;
457
458 input.y = (input.y | (input.y << 8)) & 0x00FF00FF;
459 input.y = (input.y | (input.y << 4)) & 0x0F0F0F0F;
460 input.y = (input.y | (input.y << 2)) & 0x33333333;
461 input.y = (input.y | (input.y << 1)) & 0x55555555;
462
463 res= input.x | (input.y << 1);
464
465 return res;
466 }

468 /******************************************************************************
469 * Helper function used to deinterleave an unsigned 32 bits integer
470 *
471 * @param pInput the input value
472 * @param pResult the deinterleaved output value
473 ******************************************************************************/
474 __host__ __device__
475 inline void deinterleaveBits32( const uint pInput, uint2& pResult )
476 {
477 pResult.x = pInput & 0x55555555;
478 pResult.y = ( pInput >> 1 ) & 0x55555555;
479
480 pResult.x = ( pResult.x | pResult.x>>1 ) & 0x33333333;
481 pResult.x = ( pResult.x | pResult.x>>2 ) & 0x0F0F0F0F;
482 pResult.x = ( pResult.x | pResult.x>>4 ) & 0x00FF00FF;
483 pResult.x = ( pResult.x | pResult.x>>8 ) & 0x0000FFFF;
484
485 pResult.y = ( pResult.y | pResult.y>>1 ) & 0x33333333;
486 pResult.y = ( pResult.y | pResult.y>>2 ) & 0x0F0F0F0F;
487 pResult.y = ( pResult.y | pResult.y>>4 ) & 0x00FF00FF;
488 pResult.y = ( pResult.y | pResult.y>>8 ) & 0x0000FFFF;
489 }
