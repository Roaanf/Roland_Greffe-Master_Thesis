56/******************************************************************************
57 * Constructor
58 *
59 * @param pDataStructure the data stucture to render.
60 * @param pCache the cache used to store the data structure and handle produce data requests efficiently.
61 * It handles requests emitted during rendering phase (node subdivisions and brick loads).
62 * @param pProducer the producer used to provide data following requests emitted during rendering phase
63 * (node subdivisions and brick loads).
64 ******************************************************************************/
65template< typename TDataStructureType, typename VolumeTreeCacheType >
66GsRenderer< TDataStructureType, VolumeTreeCacheType >
67::GsRenderer( TDataStructureType* pDataStructure, VolumeTreeCacheType* pCache )
68: GsIRenderer()
69, _timeBudget( 0.f )
70, _performanceTimer( NULL )
71{
72 assert( pDataStructure );
73 assert( pCache );
74
75 this->_volumeTree = pDataStructure;
76 this->_volumeTreeCache = pCache;
77
78 _updateQuality = 0.3f;
79 _generalQuality = 1.0f;
80
81 // This method update the associated "constant" in device memory
82 setVoxelSizeMultiplier( 1.0f );
83
84 _currentTime = 10;
85
86 // By default, during data structure traversal, request for load bricks strategy first
87 // (not node subdivision first)
88 _hasPriorityOnBricks = true;
89
90 // Specify clear values for the color and depth buffers
91 _clearColor = make_uchar4( 0, 0, 0, 0 );
92 _clearDepth = 1.f;
93
94 // Projected 2D Bounding Box of the GigaVoxels 3D BBox.
95 _projectedBBox = make_uint4( 0, 0, 0, 0 );
96
97 // Time budget (in milliseconds)
98 //
99 // 60 fps
100 _timeBudget = 1.f / 60.f;
101 _performanceTimer = new GvPerfMon::GsPerformanceTimer();
102 _timerEvent = _performanceTimer->createEvent();
103}
