334void GVKernel_TrackLeafNodes( TDataStructure pDataStructure, TPageTable pPageTable, const unsigned int pNbNodes, const unsigned int pMaxDepth, unsigned int* pLeafNodes, float* pEmptyVolume )
335{
336 // Retrieve global index
337 const uint index = threadIdx.x + blockIdx.x * blockDim.x + blockIdx.y * blockDim.x * gridDim.x;
338
339 // Check bounds
340 if ( index < pNbNodes )
341 {
342 // Try to retrieve node from the node pool given its address
343 GvStructure::GsNode node;
344 pDataStructure.fetchNode( node, index );
345
346 //---------------------------
347 if ( node.childAddress == 0 )
348 {
349 // Node has data
350 pLeafNodes[ index ] = 0;
351
352 // Update volume
353 pEmptyVolume[ index ] = 0.f;
354
355 return;
356 }
357 //---------------------------
358
359 // Retrieve its "node tile" address
360 //const uint nodeTileAddress = index / 8/*NodeTileRes::getNumElements()*/;
361
362 // Retrieve node depth
363 const GvCore::GsLocalizationInfo localizationInfo = pPageTable.getLocalizationInfo( index );
364 const unsigned int nodeDepth = localizationInfo.locDepth.get()/* + 1*/;
365
366 // Check node depth
367 if ( localizationInfo.locDepth.get() < pMaxDepth )
368 {
369 if ( node.childAddress != 0 )
370 {
371 // Check if node is a leaf
372 if ( ! node.hasSubNodes() )
373 {
374 // Check if node is empty
375 if ( ! node.hasBrick() )
376 {
377 // Empty node
378 pLeafNodes[ index ] = 1;
379
380 // Update volume
381 //printf( "\n%u", nodeDepth );
382 const float nodeSize = 1.f / static_cast< float >( 1 << nodeDepth );
383 pEmptyVolume[ index ] = nodeSize * nodeSize * nodeSize;
384 }
385 else
386 {
387 // Node has data
388 pLeafNodes[ index ] = 0;
389
390 // Update volume
391 pEmptyVolume[ index ] = 0.f;
392 }
393 }
394 else
395 {
396 // Node has children
397 pLeafNodes[ index ] = 0;
398
399 // Update volume
400 pEmptyVolume[ index ] = 0.f;
401 }
402 }
403 else
404 {
405 // ...
406 pLeafNodes[ index ] = 0;
407
408 // Update volume
409 pEmptyVolume[ index ] = 0.f;
410 }
411 }
412 else if ( localizationInfo.locDepth.get() == pMaxDepth )
413 {
414 if ( node.childAddress != 0 )
415 {
416 // Check if node is empty
417 if ( ! node.hasBrick() )
418 {
419 // Empty node
420 pLeafNodes[ index ] = 1;
421
422 // Update volume
423 const float nodeSize = 1.f / static_cast< float >( 1 << nodeDepth );
424 pEmptyVolume[ index ] = nodeSize * nodeSize * nodeSize;
425 }
426 else
427 {
428 // Node has data
429 pLeafNodes[ index ] = 0;
430
431 // Update volume
432 pEmptyVolume[ index ] = 0.f;
433 }
434 }
435 else
436 {
437 // ...
438 pLeafNodes[ index ] = 0;
439
440 // Update volume
441 pEmptyVolume[ index ] = 0.f;
442 }
443 }
444 else // ( localizationInfo.locDepth > pMaxDepth )
445 {
446 // Don't take node into account
447 pLeafNodes[ index ] = 0;

515{
516 //const float pi = 3.141592f;

518 // Retrieve global index
519 const uint index = threadIdx.x + blockIdx.x * blockDim.x + blockIdx.y * blockDim.x * gridDim.x;
520
521 // Check bounds
522 if ( index < pNbNodes )
523 {
524 // Try to retrieve node from the node pool given its address
525 GvStructure::GsNode node;
526 pDataStructure.fetchNode( node, index );
527
528 // Retrieve its "node tile" address
529 const uint nodeTileAddress = index / 8/*NodeTileRes::getNumElements()*/;
530
531 // Retrieve node depth
532 const GvCore::GsLocalizationInfo localizationInfo = pPageTable.getLocalizationInfo( nodeTileAddress );
533 const unsigned int nodeDepth = localizationInfo.locDepth.get() + 1;
534
535 // Check node depth
536 if ( nodeDepth < pMaxDepth )
537 {
538 if ( node.childAddress != 0 )
539 {
540 // Check if node is a leaf
541 if ( ! node.hasSubNodes() )
542 {
543 // Leaf node
544 pLeafNodes[ index ] = 1;
